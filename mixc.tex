\section{Введение}
Пусть $\Omega = \omega \times [-1,1] \subset \Real^n$, 
$F( \cdot, \cdot, \cdot ): \omega \times \Real \times \Real_+ \to \Real_+$ ---
четная, выпуклая и возрастающая по третьему аргументу непрерывная неотрицательная функция,
$a_i( \cdot, \cdot ): \omega \times \Real \to \Real$ и $a( \cdot, \cdot ): \Omega \times \Real \to \Real_+$ ---
непрерывные функции. Обозначим
$x = ( x_1, \dots, x_{n - 1}, y ) = ( x', y )$. $\norm{\cdot}$ --- некоторая
норма в $\Real^n$, симметричная по последней координате. $i$ всегда меняется от $1$ до $n - 1$.
$D_k$ будем обозначать частную производную по $k$-ой координате.
Рассмотрим измеримую функцию $u$, заданную на $\Omega$. Для нее верна формула о послойном
представлении: $u( x', y ) = \int_0^\infty \chi_{\{ y \in [-1,1] :\ u( x', y ) > t \}} dt - 
\int_{-\infty}^0 \chi_{\{ y \in [-1,1] :\ u( x', y ) < t \}} dt$.
Определим монотонную перестановку множества $E \subset [-1,1]$: $E^* = [1 - \abs{ E }, 1]$.
Тогда монотонная перестановка функции $u$ вдоль переменной $y$ определяется следующим образом:
$u^*( x', y ) = \int_0^\infty \chi_{\{ y \in [-1,1] :\ u( x', y ) > t \}^*} dt - 
\int_{-\infty}^0 1 - \chi_{\{ y \in [-1,1] :\ u( x', y ) > t \}^*} dt$.
Аналогично можно определить симметричную перестановку множества $E$ и функции $u$.
$\overline{ E } = [-\frac{\abs{E}}{2}, \frac{\abs{E}}{2}]$,
$\overline{ u }( x', y ) = \int_0^\infty \chi_{ \overline{ \{ y \in [-1, 1] :\ u( x', y ) > t \} } } dt -
\int_{-\infty}^0 1 - \chi_{ \overline{ \{ y \in [-1, 1] :\ u( x', y ) > t \} } } dt$.

Положим $u \in \W(\Omega)$. Введем следующий функционал: $$ I( u ) = \int_\Omega F( x', u(x),
\norm{ a_i( x', u( x ) ) D_i u( x ), a( x, u( x ) ) D_n u( x ) } ) dx.$$

Целью работы является доказательство неравенства
\begin{equation}
\label{toprove}
I(u^*) \le I(u).
\end{equation}

В работе \cite{Br} показано, что $I(\overline{ u }) \le I( u )$ при условии, что функция $a$ четна и выпукла по $y$.
Мы покажем, что неравенство (\ref{toprove}) выполняется, если вес $a$ четен и вогнут по $y$ и в некоторых случаях
можно ослабить условие вогнутости.

Не умаляя общности, можно считать, что $u( x ) \ge 0$ всюду.
В работе \cite{Lan} подобное неравенство доказывается для функций $u$,
закрепленных на левом конце: $u( -1 ) = 0$, при условии, что вес убывает.

\section{Необходимые условия}

Будем рассматривать условия, необходимые для выполнения
неравенства (\ref{toprove}) в одномерном случае. Также можно
считать, что $F$ зависит только от последнего аргумента, а $a$ ---
только от первого. Такие условия будут необходимыми и в
общем случае.

Пусть $u$ убывает. Тогда
\begin{multline*}
I(u)-I(u^*) =
\int_{-1}^1 (
    F( \abs{ a(x) u'(x) } ) -
    F( \abs{ a(x) u'(-x) } )
) dx = \\ =
\int_{-1}^1 (
    F( \abs{ a(x) u'(x) } ) -
    F( \abs{ a(-x) u'(x) } )
) dx
\end{multline*}

\begin{prop}
$a$ должна быть четной, то есть $\forall x\ a(x)=a(-x)$
\end{prop}

\begin{proof} Пусть $a(\bar{x})<a(-\bar{x})$, $\abs{\bar{x}} \neq 1$. Тогда
$$\exists \eps : |x-\bar{x}| < \eps \Rightarrow a(x) < a(-x)$$
и можно взять следующую функцию: $u([-1,\bar{x}-\eps]) = 1$,
$u([\bar{x}+\eps, 1]) = 0$,\newline
$u(x)=-\frac{x}{2\eps}+\frac{\bar{x}+\eps}{2\eps}$ при
$|x-\bar{x}|<\eps$. Она убывает. Значит
$$I(u)-I(u^*) =
\int_{|x-\bar{x}|>\eps} (
    F( 0 ) - F( 0 )
) dx +
\int_{\bar{x}-\eps}^{\bar{x}+\eps} (
    F( \frac{ a(x) }{ 2\eps } ) -
    F( \frac{ a(-x) }{ 2\eps } )
) dx <0.$$
 Тем самым мы пришли к противоречию.
\end{proof}

\begin{prop}
$\forall s \le t \ a( s ) + a( t ) \ge a( 1 - t + s ).$
\end{prop}

\begin{proof}
Предположим, для некоторых $-1 < s \le t < 1$ неравенство неверно:
$a( s ) + a( t ) + 2 \delta < a( -1 + t - s )$. Тогда найдется
$\eps > 0$ такое, что для произвольного $0 \le y \le \eps$ останется
$a( s - y ) + a( t + y ) + \delta < a( -1 + t - s + 2y )$.
Построим функцию $u$, имеющую ненулевые производные только в окрестностях $s$ и $t$.
$u( [-1,s] \cup [t,1] ) = 0$, $u( [s+\eps,t-\eps] ) = 1$,
$u( x ) = \frac{ x - s }{ \eps }$ при $x \in [s,s+\eps]$,
$u( x ) = \frac{ t - x }{ \eps }$ при $x \in [t-\eps,t]$.
Тогда $u^*( [-1, 1 - t + s] ) = 0$, $u^*( [1 - t + s + 2\eps, 1] ) = 1$,
$u^*( x ) = \frac{ x - ( 1 - t + s ) }{2 \eps}$ при $x \in [1 - t + s, 1 - t + s + 2\eps]$.
Тогда получим
\begin{multline*}
0 \le I( u ) - I( u^* ) =
\int_0^\eps F( \frac{a( s + z )}{\eps} ) + F( \frac{ s( t - z ) }{\eps} ) - F( \frac{ a( 1 - t + s + 2 z ) }{2 \eps} ) <
\\ \int_0^\eps F( \frac{a( s + z )}{\eps} ) + F( \frac{ s( t - z ) }{\eps} ) -
F( \frac{ a( s - z ) + a( t + z ) + \delta }{2 \eps} ) =: J
\end{multline*}

Положим теперь $F(x) = x^\alpha$. Такая функция строго выпукла при $\alpha > 1$. Однако по заданному $\delta$ можно
найти такое $\alpha$, чтобы $\frac{ F( x ) + F( y ) }{2} \le F( \frac{x + y}{2} + \frac{\delta}{2 \eps} )$ при
$x,y \in [0,A]$. Для этого решим неравенство
$\frac{ x^\alpha + y^\alpha }{2} \le ( \frac{x + y}{2} + \frac{\delta}{2 \eps} )^\alpha$
относительно $\frac{\delta}{2 \eps}$. Получим
$\frac{\delta}{2 \eps} \ge ( \frac{ x^\alpha + y^\alpha }{2} )^{\frac{1}{\alpha}} - \frac{x + y}{2}$.
Максимум последнего выражения достигается, например, при $x = 0$, $y = A$, то есть
$\frac{\delta}{2 \eps} \ge A(\frac{1}{2^{\frac{1}{\alpha}}} - \frac{1}{2})$. Итого, можно взять
$\alpha = -( \log_2 \frac{\delta}{2 A \eps} + \frac{1}{2} )^{-1} > 1$.

Тогда, если взять $A = \max a$, $\alpha = -( \log_2 \frac{\delta}{2 A \eps} + \frac{1}{2} )^{-1}$
и $F( x ) = x^\alpha$, получим $J \le 0$ и противоречие.
\end{proof}

\rem Из доказательства ясно, что последнее условие является необходимым и для выполнения неравенства (\ref{toprove})
в случае закрепленных на левом конце функций: $u( -1 ) = 0$.

\section{Свойства весовой функции}

\begin{lm}
\label{weightSum}
Рассмотрим непрерывную функцию $a \ge 0$, заданную на $[-1,1]$ и для произвольных $s \le t \in [-1,1]$
удовлетворяющую условию
\begin{equation}
\label{almostConcave}
a( s ) + a ( t ) \ge a( 1 - t + s ).
\end{equation}
Тогда для любых $-1 \le t_1 \le t_2 \le \ldots \le t_n \le 1$ верно
\begin{align*}
\sum_{k=1}^n a(t_k) & \ge a( 1 - \sum_{k = 1}^n (-1)^k t_k ), & \text{ для четных $n$},&\\
\sum_{k=1}^n a(t_k) & \ge a( -\sum_{k = 1}^n (-1)^k t_k ),    & \text{ для нечетных $n$}.&
\end{align*}
\end{lm}

\begin{proof}
Будем доказывать по индукции. Для $n = 1$ утверждение тривиально.
Пусть теперь $n$ четное. Тогда, по предположению индукции,
$\sum_{k=1}^{n - 1} a(t_k) \ge a( -\sum_{k = 1}^{n - 1} (-1)^k t_k )$. Значит
$$\sum_{k = 1}^{n - 1} a( t_k ) + a( t_n ) \ge a( -\sum_{k = 1}^{n - 1} (-1)^k t_k ) + a( t_n ) \ge
a( 1 - \sum_{k = 1}^{n} (-1)^k t_k ).$$
В случае нечетного $n$ воспользуемся предположением индукции в следующем виде:
$\sum_{k=2}^n a(t_k) \ge a( 1 + \sum_{k = 2}^n (-1)^k t_k )$. Тогда
$$a( t_1 ) + \sum_{k = 2}^{n - 1} a( t_k ) \ge a( t_1 ) + a( 1 + \sum_{k = 2}^{n} (-1)^k t_k ) \ge
a( -\sum_{k = 2}^{n} (-1)^k t_k + t_1 ) = a( -\sum_{k = 1}^{n} (-1)^k t_k ).$$
\end{proof}

\rem Если вдобавок к условию леммы предположить, что функция $a$ четна, то выполняются также следующие неравенства:
\begin{align*}
\sum_{k=1}^n a(t_k) & \ge a( -1 + \sum_{k = 1}^n (-1)^k t_k ), & \text{ для четных $n$},&\\
\sum_{k=1}^n a(t_k) & \ge a( \sum_{k = 1}^n (-1)^k t_k ),    & \text{ для нечетных $n$}.&
\end{align*}

\rem Для четной функции $a$ условие (\ref{almostConcave}) можно переписать в следующем виде: для любых $s,t \in [-1,1]$ таких, что
$1 + s + t \le 1$ верно 
\begin{equation}
\label{almostConcaveSum}
a( s ) + a( t ) \ge a( 1 + s + t ).
\end{equation}

\begin{lm}
\label{periodicity}
Если четная функция $a$ удовлетворяет условию (\ref{almostConcaveSum}), а также найдется такое $x_0 > -1$, что
$a(x_0) = 0$, то $a(-1) = 0$ и функция $a$ имеет период $x_0+1$. В частности, для любого неотрицательного
целого $k$ такого, что $-1 + k (x_0 + 1) \le 1$, выполнено $a(-1 + k (x_0 + 1)) = 0$.
\end{lm}

\begin{proof}
Возьмем в условии (\ref{almostConcaveSum}) $s = x_0$, $t = -x_0$. Тогда
$a(s) + a(t) = 0 \ge a(1 + s + t) = a(1) \ge 0$. Значит $a(-1) = a(1) = 0$.
Возьмем некоторое $x \in [-1,1]$ и докажем, что $a(x) = a( x + (1 + x_0) )$ при условии, что $1 + x + x_0 <= 1$.
В условии (\ref{almostConcaveSum}) положим $s = x_0$, $t = x$. Тогда $a(x) \ge a( x + (1 + x_0) )$.
Теперь положим $s = x_0$, $t = -x - (1 + x_0)$. Тогда $a(t) = a(x + (1 + x_0)) \ge a(-x) = a(x)$. Значит
$a(x) = a( x + (1 + x_0) )$.
\end{proof}

\begin{lm}
\label{piecewiseLinearConcave}
Пусть функция $a$ удовлетворяет неравенству (\ref{almostConcaveSum}), $k \in \Nat$.
Тогда кусочно-линейная функция $a_k$, построенная на точках
$(-1 + \frac{2i}{k}, a(-1 + \frac{2i}{k}))$ при $i = 0, 1, \dots, k$, тоже
удовлетворяет (\ref{almostConcaveSum}).
\end{lm}
\begin{proof}
Пусть $s = -1 + \frac{2i}{k}$, $t = -1 + \frac{2j}{k}$. Тогда неравенство
выполняется для $a_k$, потому что оно выполняется для $a$. Пусть теперь для
некоторого $s = -1 + \frac{2i}{k}$ и некоторого
$t \in [-1 + \frac{2j}{k}, -1 + \frac{2(j + 1)}{k}]$
$a_k(s) + a_k(t) < a_k( 1 + s + t )$. Рассмотрим линейную функцию
$h(t) = a_k( 1 + s + t ) - a_k(t) - a_k(s)$.
$h(-1 + \frac{2j}{k}) \le 0$, $h(t) > 0$, значит
$h(-1 + \frac{2(j + 1)}{k}) > 0$, что противоречит уже доказанному.
Тем самым, неравенство выполняется для любого $s \in -1 + \frac{2i}{k}$
и $t \in [-1, 1]$. Аналогично, заведя линейную функцию
$g(s) = a_k( 1 + s + t ) - a_k(t) - a_k(s)$ для произвольного
фиксированного $t \in [-1, 1]$, и предположив, что $g(s) > 0$ для некоторого
$s \in [-1 + \frac{2i}{k}, -1 + \frac{2(i + 1)}{k}]$, придем к противоречию
и получим неравенство для любых $s$ и $t$.
\end{proof}

\begin{lm}
\label{maxConcave}
Пусть функции $a_1$ и $a_2$ удовлетворяют неравенству (\ref{almostConcaveSum}),
тогда функция $a(x) = \max (a_1(x), a_2(x))$ тоже ему удовлетворяет.
\end{lm}
\begin{proof}
$a(1 + s + t ) = \max( a_1(1 + s + t), a_2(1 + s + t) ) \le
\max( a_1( s ) + a_1( t ), a_2( s ) + a_2( t ) ) \le
\max( a_1( s ), a_2( s ) ) + \max( a_1( t ), a_2( t ) ) =
a( s ) + a( t ).$
\end{proof}

Рассмотрим кусочно монотонную функцию $a$. Возьмем некоторое число $\gamma > 0$
и заменим функцию $a$ в окрестностях ненулевых минимумов на константу так, чтобы
новая функция $a_\gamma$ оставалась непрерывной и значение в каждом из минимумов
поднялось на $\gamma$. Очевидно, при достаточно малой длине отрезков постоянства,
а значит, и при достаточно малом $\gamma$, функция $a_\gamma$ определена.
Видно, что монотонность в каждой точке сохраняется, становясь в окрестностях
ненулевых минимумов нестрогой, что даст нам в дальнейшем свободу выбора
границ участков монотонности.

\begin{lm}
\label{flattenMin}
Пусть функция $a$ кусочно монотонна и удовлетворяет неравенству (\ref{almostConcaveSum}),
тогда найдется число $\gamma_0 > 0$ такое, что при любом $\gamma \le \gamma_0$
функция $a_\gamma$ удовлетворяет неравенству (\ref{almostConcaveSum}).
\end{lm}
\begin{proof}
Введем обозначение для множества нулей функции $a$: $Z = a^{-1}(0)$ и для
множества локальных строгих ненулевых минимумов функции $a$ за исключением
крайних точек отрезка
$M = \{ x \in ( -1, 1 ) \setminus Z: \exists \eps( x ) \forall y :
0 < \abs{ x - y } < \eps \ a( y ) > a( x ) \}$.
Докажем утверждение леммы от противного.
Пусть $\gamma_i \to 0$ и для каждого $i$ найдутся $s_i$ и $t_i$, что
$a_{\gamma_i}( s_i ) + a_{\gamma_i}( t_i) < a_{\gamma_i}( 1 + s_i + t_i )$.
Поскольку $( s_i, t_i )$ лежат в компакте, можно, перейдя к
подпоследовательности, получить $s_i \to s$ и $t_i \to t$.
Поскольку $a_{\gamma_i}( x ) \to a( x )$, в пределе получим
$a( s ) + a( t ) = a( 1 + s + t )$.
Докажем от противного, что $1 + s + t \in M$ (и, в частности,
$1 + s + t \in (-1, 1)$). Пусть $1 + s + t \notin M$. Тогда, начиная с
некоторого $i$, будет $a_{\gamma_i}( 1 + s_i + t_i ) = a( 1 + s_i + t_i ) \le
a( s_i ) + a( t_i ) \le a_{\gamma_i}( s_i ) + a_{\gamma_i}( t_i ) <
a_{\gamma_i}( 1 + s_i + t_i )$. Теперь докажем от противного, что
$s, t \in M \cup Z$. Пусть, например, $s \notin M \cup Z$. Тогда
найдется последовательность $\delta _i \to 0$, что
$a( s + \delta _i ) \le a( s )$. Но тогда $a( s + \delta _i ) + a( t )
\ge a( 1 + s + \delta _i + t ) > a( 1 + s + t ) = a( s ) + a( t ) \ge
a( s + \delta _i ) + a( t )$. Здесь $1 + s + \delta _i + t \in (-1, 1)$
начиная с некоторого $i$, так как $1 + s + t \in M$. По той же причине
выполняется второе неравенство.
Таким образом, $s, t \in M \cup Z, 1 + s + t \in M$.
Пусть $s, t \in Z$, тогда $1 + s + t \in Z$, но такого быть не может,
поэтому либо $s$, либо $t$ находится в $M$.
Если $s, t \in M$, получаем $( a( s ) + \gamma_i ) + ( a( t ) + \gamma_i ) =
a_{\gamma_i}( s_i ) + a_{\gamma_i}( t_i ) < a_{\gamma_i}( 1 + s_i + t_i ) =
a( 1 + s + t ) + \gamma_i$, то есть
$a( s ) + a( t ) < a( 1 + s + t )$. Если же, например, $s \in Z, t \in M$, то
$a( s ) + ( a( t ) + \gamma_i ) =
a_{\gamma_i}( s_i ) + a_{\gamma_i}( t_i ) < a_{\gamma_i}( 1 + s_i + t_i ) =
a( 1 + s + t ) + \gamma_i$, то есть опять же
$a( s ) + a( t ) < a( 1 + s + t )$
Тем самым, мы пришли к противоречию,
доказав для достаточно малых $\gamma$ $a_\gamma( s ) + a_\gamma( t ) \ge a_\gamma( 1 + s + t )$.
\end{proof}

\section{Результат на кусочно-линейных функциях}

Рассмотрим случай, когда $a$ не зависит от $u$. Докажем наше неравенство для кусочно-линейных функций,
разрешив им иметь разрывы в множестве $Z = a^{-1}(0)$ нулей функции $a$. При интегрировании будем
считать вклад этих точек нулевым.

\begin{thm}
\label{linth}
Рассмотрим одномерный случай: $n = 1$, $\Omega = [-1, 1]$.
Пусть функция $a$ зависит только от $x$, неотрицательна, четна и удовлетворяет условию (\ref{almostConcave}).
Тогда, если $u$ --- кусочно-линейная функция с разрывами лишь в точках множества $Z$, то функция $u^*$
тоже будет иметь разрывы только на множестве $Z$ и $I(u)\ge I(u^*)$.
\end{thm}

\begin{proof}
Положим $-1 = x_1 < x_2 < \dots < x_K = 1$ --- множество точек перелома функции $u$, включая точки
разрывов. Рассмотрим множество $U$ образов функции $u$, не являющихся образами конечных точек
линейных участков: $U := u( \Omega ) \setminus \cup_{i = 2}^K \{ \lim\limits_{x \nearrow x_i} u(x) \}
\setminus \cup_{i = 1}^{K - 1} \{ \lim\limits_{x \searrow x_i} u(x) \}$. Множество $U$ можно
разбить на конечное число связных областей $U = \cup_{j = 1}^N G_j$. При $u_0 \in G_j$ число решений уравнения
$u(y) = u_0$ не зависит от $u_0$ и в этой области задается набором линейных функций: $y_k^j(u_0) = y$,
$k = 1 \dots m_j$ (расположим их в порядке увеличения значений). При этом будем
считать, что каждому разрыву соответствует вертикальный участок графика, а
также некоторое решение $y_k^j(u)$, постоянное по $u$. При $y \in u^{-1}(U)$
$y_k^j{}'(u(y))=\frac{1}{u'(y)}$, если в точке $y$ разрыв, то считаем
$y_k^j{}'(u(y))=0$.

Решение уравнения $u^*(y^*)=u_0$ ($u_0 \in U$) можно выразить используя решения $y_k^j$.
Если $u(-1)<u_0$ и $m_j$ четное, то $y^*=1-\sum_{k=1}^{m_j} (-1)^k y_k^j$, если нечетное, то
$y^*=-\sum_{k=1}^{m_j} (-1)^k y_k^j$, а в случае $u(-1)>u_0$ $y^*=-1+\sum_{k=1}^{m_j} (-1)^k y_k^j$
при четном $m_j$ и $y^*=\sum_{k=1}^{m_j} (-1)^k y_k^j$ при нечетном.
Будем за $y^*$ обозначать $(u^*)^{-1}$ (при этом разрывы переходят в константные участки и наоборот).
Тогда ${y^*}'(u) = \sum_{k=1}^{m_j} \abs{y_k^j{}'(u)}$ при $u \in G_j$, поскольку знаки в выражении для
$y^*$ и знаки $y_k^j$ чередуются и ${y^*}'(u)\ge 0$.
Отсюда ясно, что константный кусок у
$y^*( u )$ может быть только при тех $u$, при которых все $y_k^j(u)$ константные. То есть все
$y_k^j(u)$ --- точки разрыва, а значит расстояние от $-1$ до каждой из них кратно периоду. Тогда ясно, что
расстояние от $-1$ до $y^*(u)$ тоже кратно периоду, то есть $y^*(u) \in Z$. Тем самым, $u^*$ может иметь
разрывы лишь в точках множества $Z$. Перейдем к доказательству неравенства.

Множества нулей $u'$ и ${u^*}'$ могут иметь ненулевую меру. Однако, их меры
совпадают. Более того, меры множеств их нулей при фиксированном значении
функции тоже совпадают. Тогда
$$\int_{(u')^{-1} (0)} F( u(x), 0 ) dx = \int_{({u^*}')^{-1} (0)} F( u^*(x), 0 ) dx =: C.$$
Рассмотрим оставшиеся части интегралов:
\begin{multline*}
I(u) - C = \sum_{j=1}^N \int_{u^{-1}(G_j)} F( u(x), \abs{ a(x) u'(x) } ) dx =
\\ \sum_{j=1}^N \int_{G_j} \sum_{k=1}^{m_j} F( u, \frac{ a(y_k^j(u)) }{ \bigabs{y_k^j{}'(u)} } ) \bigabs{y_k^j{}'(u)} du,
\end{multline*}
\begin{multline*}
I(u^*) - C = \sum_{j=1}^N \int_{(u^*)^{-1}(G_j)} F( u^*(x), \bigabs{ a(x) {u^*}'(x) } ) dx = 
\\ \sum_{j=1}^N \int_{G_j} F( u^*, \frac{ a(y^*(u^*)) }{\sum_{k=1}^{m_j} \bigabs{ y_k^j{}'(u^*) }} )
\sum_{k=1}^{m_j} \bigabs{ y_k^j{}'(u^*) } du^*.
\end{multline*}
Зафиксируем $j$ и $u = u^*$ в правых частях и докажем
неравенство для подынтегральных выражений. Обозначим
$b_k=|y_k^j{}'(u)|$, $y_k=y_k^j(u)$, $y^*=y^*(u)$, $m = m_j$. Тогда
требуемое утверждение имеет вид:
$$T:=\sum_{k=1}^m b_k F( \frac{ a(y_k) }{b_k} ) \ge F( \frac{ a(y^*) }{ \sum_{k=1}^m b_k  } ) \sum_{k=1}^m b_k$$
С помощью неравенства Йенсена для функции $F$ получаем
$$T \ge F( \frac{ \sum_{k=1}^m a(y_k) }{ \sum_{k=1}^m b_k } ) \sum_{k=1}^m b_k.$$
Тогда нам достаточно установить $\sum_{k=1}^m a(y_k) \ge a(y^*)$, что верно с учетом леммы \ref{weightSum} и замечания
после нее.
\end{proof}

\rem Если положить $u( -1 ) = \min\limits_\Omega u$, то леммы \ref{weightSum} будет достаточно для выполнения теоремы, поскольку
$u( -1 ) < u_0$ и нам понадобятся только два неравенства из четырех, как раз те, которые обеспечивает лемма.
В работе \cite{Lan} результат доказывается для убывающих функций $a$. Несложно заметить, что
условие леммы выполняется для убывающих функций, поскольку $s \le 1 - t + s$, а значит и
$a( s ) + a( t ) \ge a( 1 - t + s )$.

\section{Слабая полунепрерывность функционала}

\begin{prop}
Функция $a$ непрерывна. Тогда функционал $I(u)$ слабо полунепрерывен снизу в $\W(\Omega)$.
\label{lowersemi}
\end{prop}

\begin{proof}
Пусть $u_m \rightharpoondown u$ в $\W$. Обозначим $A = \varliminf I( u_m ) \ge 0$. Наша задача --- доказать
$I(u) \le A$. Если $A = \infty$, то утверждение тривиально, так что можно считать $A < \infty$.
Переходя к подпоследовательности, добиваемся $A = \lim I( u_m )$. Из слабой сходимости заключаем, что найдется
$R_0$ такое, что $\norm{ u_m }_{\W(\Omega)} \le R_0$.

Известно, что $\W(\Omega)$ ограниченно и компактно вкладывается в $L_1(\Omega)$. Тогда, переходя к подпоследовательности,
$u_m \to u$ в $L_1(\Omega)$. Тогда, опять по подпоследовательности, $u_m(x) \to u(x)$ почти всюду.
Тогда по теореме Егорова для любого $\eps$ найдется
$G_\eps^1$ такое, что $\abs{ G_\eps^1 } < \eps$ и $u_m \rightrightarrows u$ в $\Omega \setminus G_\eps^1$.
Из равномерной сходимости $\exists K: \forall k>K \ \abs{u_k} \le \abs{u} + \eps$ в $\Omega \setminus G_\eps^1$.
Возьмем $G_\eps^2 = \{ x \in \Omega \setminus G_\eps^1: \exists k > K \ \abs{u_k(x)} \ge \frac{R_0}{\eps} + \eps \}$.
Тогда $$R_0 \ge \int_\Omega \abs{ u(x) } dx \ge \int_{G_\eps^2} \abs{ u(x) } dx \ge
\int_{G_\eps^2} ( \abs{u_k(x)} - \eps ) dx \ge \int_{G_\eps^2} \frac{R_0}{\eps} dx
= \abs{G_\eps^2} \frac{R_0}{\eps}$$
То есть $\abs{G_\eps^2} \le \eps$.
Таким образом, вне множества
$G_\eps = G_\eps^1 \cup G_\eps^2$
функции $u_m$ равномерно по $m$ ограничены и равномерно по $x$ сходятся к $u$. При этом $\abs{ G_\eps } < 2\eps$.

Из непрерывности $F$, $a_i$ и $a$, а также равномерной сходимости и ограниченности $u_m$ вне множеств $G_\eps$,
следует, что для произвольных $\eps$ и $R$ найдется
$N( \eps, R )$, что если $x \in \Omega \setminus G_\eps$, $\abs{ M_i } \le R$, $\abs{ M } \le R$ и $m > N( \eps, R )$, то
\begin{multline*}
| F( x', u_m( x ), \norm{ a_i( x', u_m( x ) ) M_i, a( x, u_m( x ) ) M } ) -
\\ F( x', u( x ), \norm{ a_i( x', u( x ) ) M_i, a( x, u( x ) ) M } ) | < \eps.
\end{multline*}

Выделим множества $E^1_{m,\eps} = \{ x \in \Omega: \norm{ \grad u_m( x ) } \ge \frac{ R_0 }{ \eps } \}$. Тогда
$$R_0 \ge \int_\Omega \norm{ \grad u_m( x ) } dx \ge \int_{ E^1_{m,\eps} } \norm{ \grad u_m( x ) } dx \ge
\int_{ E^1_{m,\eps} } \frac{ R_0 }{ \eps } dx = \frac{ R_0 }{ \eps } \abs{ E_{m,\eps} }.$$ А значит,
$\abs{ E^1_{m,\eps} } \le \eps$. Теперь можно ввести $E_{m,\eps} = G_\eps \cup E^1_{m,\eps}$ и
$L_{m,\eps} = \Omega \setminus E_{m,\eps}$. Тогда $\abs{ E_{m,\eps} } \le 3 \eps$.

Зафиксируем $R = \frac{ R_0 }{ \eps }$. Теперь $N( \eps ) = N( \eps, R_0 )$.
Для любых $\eps > 0$, $x \in L_{m,\eps}$ и $m > N( \eps )$ получим
\begin{multline*}
| F( x', u_m( x ), \norm{ a_i( x', u_m( x ) ) D_i u_m( x ), a( x, u_m( x ) ) D_n u_m( x ) } ) -
\\ F( x', u( x ), \norm{ a_i( x', u( x ) ) D_i u_m( x ), a( x, u( x ) ) D_n u_m( x ) } ) | < \eps.
\end{multline*}
Или
\begin{multline*}
\int_{L_{m,\eps}} | F( x', u_m( x ), \norm{ a_i( x', u_m( x ) ) D_i u_m( x ), a( x, u_m( x ) ) D_n u_m( x ) } ) -
\\ F( x', u( x ), \norm{ a_i( x', u( x ) ) D_i u_m( x ), a( x, u( x ) ) D_n u_m( x ) } ) | dx < \abs{\Omega} \eps.
\end{multline*}
Возьмем $\eps_j = \frac{ \eps }{ 2^j }$, $m_j = N( \eps_j ) + j \to \infty$ и $L_\eps = \bigcap L_{m_j,\eps_j}$.
Тогда $\sum \eps_j = \eps$ и $\abs{ \Omega \setminus L_\eps } < 3 \eps$.
Теперь можно заключить, что
\begin{multline*}
\int_{L_\eps} | F( x', u_{m_j}( x ),
\norm{ a_i( x', u_{m_j}( x ) ) D_i u_{m_j}( x ), a( x, u_{m_j}( x ) ) D_n u_{m_j}( x ) } ) -
\\ F( x', u( x ), \norm{ a_i( x', u( x ) ) D_i u_{m_j}( x ), a( x, u( x ) ) D_n u_{m_j}( x ) } ) | dx < \abs{\Omega} \eps_j.
\end{multline*}
\begin{multline*}
A = \lim I( u_{m_j} ) = \\ \lim \int_\Omega F( x', u_{m_j}( x ),
\norm{ a_i( x', u_{m_j}( x ) ) D_i u_{m_j}( x ), a( x, u_{m_j}( x ) ) D_n u_{m_j}( x ) } ) dx \ge
\\ \varliminf \int_\Omega \chi_{L_\eps} ( x ) ( F( x', u( x ),
\norm{ a_i( x', u( x ) ) D_i u_{m_j}( x ), a( x, u( x ) ) D_n u_{m_j}( x ) } ) + 
\\ ( F( x', u_{m_j}( x ),
\norm{ a_i( x', u_{m_j}( x ) ) D_i u_{m_j}( x ), a( x, u_{m_j}( x ) ) D_n u_{m_j}( x ) } ) -
\\ F( x', u( x ), \norm{ a_i( x', u( x ) ) D_i u_{m_j}( x ), a( x, u( x ) ) D_n u_{m_j}( x ) } ) ) )dx = \\
\varliminf \int_\Omega \chi_{L_\eps} ( x ) F( x', u( x ),
\norm{ a_i( x', u( x ) ) D_i u_{m_j}( x ), a( x, u( x ) ) D_n u_{m_j}( x ) } ) dx = \\
\varliminf J_\eps( \grad u_{m_j} )
\end{multline*}
Наш новый функционал
$$J_\eps( v ) = \int_\Omega \chi_{L_\eps} ( x ) F( x', u( x ),
\norm{ a_i( x', u( x ) ) v_i( x ), a( x, u( x ) ) v_n( x ) } ) dx$$
выпуклый. Перейдем к подпоследовательности последовательности $u_{m_j}$, обозначив ее $u_k$, так, что на ней
$\varliminf J_\eps( \grad u_{m_j} ) = \lim J_\eps( \grad u_k )$. Так как $\grad u_k \rightharpoondown \grad u$ в $L_1$,
то можно подобрать последовательность
выпуклых комбинаций $\grad u_k$, которые будут сходиться к $\grad u$ сильно. А именно: найдутся $\alpha_{k,l} \ge 0$ для
$k \in \Nat$, $l \le k$ такие, что $\sum_{l = 1}^k \alpha_{k,l} = 1$ для каждого $k$ и 
$w_k = \sum_{l = 1}^k \alpha_{k,l} \grad u_{k + l} \to \grad u$ в $L_1$. %Покажем, что
Так как $J_\eps$ выпуклый, $$A \ge \lim J_\eps( \grad u_k ) = \lim \sum_{l = 1}^k \alpha_{k,l} J_\eps( \grad u_{k + l} ) \ge
\varliminf J_\eps( w_k ) = J_\eps( \grad u ).$$

Поясним последнее равенство. $w_k = \sum_{l = 1}^k \alpha_{k,l} \grad u_{k + l} \to \grad u$ в $L_1$. Тогда, по
подпоследовательности, $\varliminf J_\eps(w_k) = \lim J_\eps(w_k)$ и  $w_k( x ) \to \grad u( x )$ почти всюду.
Кроме того, так как в $L_\eps$ $\abs{ \grad u_j( x ) } < \frac{ R_0 }{\eps}$, то и $\abs{ w_k( x ) } < \frac{ R_0 }{\eps}$.
Значит 
\begin{multline*}
F( x', u( x ), \norm{ a_i( x', u( x ) ) w_k^i( x ), a( x, u( x ) ) w_k^n( x ) } ) \le
\\ \max F( x', u( x ), \norm{ a_i( x', u( x ) ) M_i, a( x, u( x ) ) M } ) < \infty,
\end{multline*}
где максимум берется по компактному множеству
$( M_i, M, x ) \in [-\frac{ R_0 }{\eps},\frac{ R_0 }{\eps}] ^n \times \Omega$ от непрерывной функции.
Таким образом, применима теорема Лебега, и мы получаем необходимое равенство. Осталось заметить, что при $\eps \to 0$
$J_\eps( u ) \to I( u )$ из абсолютной непрерывности интеграла. Тем самым, утверждение доказано.
\end{proof}

\rem Рассмотрим одномерный случай ($n = 1$) и предположим, что $a$ зависит только от $x$. Тогда функционал
$I(u)$ слабо полунепрерывен снизу в $\W(\Omega \setminus Z)$ ($Z = a^{-1} (0)$).

\section{О расширении класса функций, для которых выполняется $I(u^*) \le I(u)$}
\begin{prop}
\label{uplift}
Пусть $A \subset \W(-1,1)$. И пусть $B \subset A$ таково, что
$\forall v \in B$ $I(v^*) \le I(v)$. Предположим, что для каждого $u \in A$
найдется последовательность $u_m \in B$ такая, что $u_m \to u$ в $\W(-1, 1)$ и
$I(u_m) \to I(u)$. Тогда $\forall u \in A$ $I(u^*) \le I(u)$.
\end{prop}
\begin{proof}
Возьмем некоторую $u \in A$ и для нее найдем соответствующие $u_m \in B$.
По условию $I(u_m^*) \le I(u_m) \to I(u)$.
В \cite[Theorem 1]{Br} показано, что из $u_m \to u$ в $\W(-1,1)$ следует
$\overline{u_m} \rightharpoondown \overline{u}$ в $\W(-1,1)$. Но
$u_m^*( x ) = \overline{u_m}( \frac{x - 1}{2} )$ и
$u^*( x ) = \overline{u}( \frac{x - 1}{2} )$. Значит, и
$u_m^* \rightharpoondown u^*$. Тогда из слабой полунепрерывности снизу
функционала $I$ заключаем $I(u^*) \le \liminf I(u_m^*)$. Тем самым, получаем
$I(u^*) \le I(u)$.
\end{proof}

\section{Переход к липшицевым функциям}

\begin{prop}
Пусть вес $a$ непрерывен и для любой кусочно-линейной функции $u$ $I(u^*) \le I(u)$
Тогда для любой липшицевой функции $u$ верно $I(u^*) \le I(u)$.
\end{prop}

\begin{proof}
Известно, что $\forall \eps \exists v_\eps \in C^1(\Omega): \abs{ \set{ x : u( x ) \neq v_\eps ( x ) } } < \eps$
(см. например \cite[3.1.15]{Fed}).
Возьмем тогда $v_j = v_{\frac{1}{2^j}}$. Тогда
$\abs{ \set{ x: \exists k \ge j : u( x ) \neq v_k( x ) } } < \frac{1}{2^{j - 1}}$. То есть
$v_j \to u$ почти всюду в $\Omega$. Кроме того, почти всюду, где $u( x ) = v_j( x )$, $u'( x ) = v_j'( x )$.
Таким образом, $v_j' \to u'$ почти всюду в $\Omega$.

Покажем теперь, что для каждой функции $v \in C^1( \Omega )$ можно найти кусочно-линейные $w_j$ такие, что
$w_j \to v$ и $w_j' \to v'$ почти всюду в $\Omega$.

Фиксируем $j$ и рассмотрим разбиение $\Omega$ на одинаковые параллелепипеды с диаметром, не превышающим $\frac{1}{j}$.
Разобьем каждый из них одинаковым способом на $2^n - n$ симплексов, которые обозначим $T_{jk}$.
Тем самым $\Omega = \cup_k T_{jk}$. Обозначим вершины $T_{jk}$ $\set{t_{jkl}}_{l = 0}^{n}$. Тогда для любых
$j$, $k$ набор $\set{ e_{jkl} = \frac{t_{kjl} - t_{jk0}}{\abs{t_{kjl} - t_{jk0}}} }_{l = 1}^n$
образует нормированный базис в $\Real^n$. Пусть $x \in T_{jk}$ и $x = \sum \alpha_l t_{jkl}$,
положим $w_j( x ) = \sum \alpha_l v( t_{jkl} )$. Тогда
\begin{multline}
\label{WNormEstim}
\int_\Omega \norm{ v'( x ) - w_j'( x ) } dx = \sum_k \int_{T_{jk}} \norm{ v'( x ) - w_j'( x ) } dx \le
\\ \sum_k \int_{T_{jk}} C \sum_{l = 1}^n \abs{ ( v'( x ) - w_j'( x ) ) e_{jkl} } dx =
\\ \sum_k \int_{T_{jk}} C \sum_{l = 1}^n \abs{ \frac{ \partial v }{ \partial e_{jkl} } ( x ) -
\frac{ v( t_{jkl} ) - v( t_{jk0} ) }{ \abs{ t_{jkl} - t_{jk0} } } } dx =
\\ \sum_k \int_{T_{jk}} C \sum_{l = 1}^n \abs{ \frac{ \partial v }{ \partial e_{jkl} } ( x ) -
\frac{ \partial v }{ \partial e_{jkl} } ( x_0 ) } dx \le
\\ \sum_k \int_{T_{jk}} C \sum_{l = 1}^n \max_{ \abs{x_1 - x_2} \le \frac{1}{j} }
\abs{ \frac{ \partial v }{ \partial e_{jkl} } ( x_1 ) - \frac{ \partial v }{ \partial e_{jkl} } ( x_2 ) } dx \le
\\ C \sum_{l = 1}^n \max_{ \abs{x_1 - x_2} \le \frac{1}{j} } \norm{ v'( x_1 ) - v'( x_2 ) } \sum_k \int_{T_{jk}} dx =
C n \abs{ \Omega } \max_{ \abs{x_1 - x_2} \le \frac{1}{j} } \norm{ v'( x_1 ) - v'( x_2 ) } \to 0,
\end{multline}
где $x_0$ --- некоторая точка между $t_{jk0}$ и $t_{jkl}$. Поскольку использованных базисов лишь конечное
количество, существует общая константа $C$. Очевидно, что
$$\abs{ v( x ) - w_j( x ) } \le 2 \max_{ \abs{x_1 - x_2} \le \frac{1}{j} } \abs{ v( x_1 ) - v( x_2 ) } \to 0.$$
Таким образом, $w_j$, а также их частные производные (как следует из вычислений (\ref{WNormEstim})) почти всюду
сходятся к $v$ и ее частным производным. Кроме того, $w_j \to v$ в $\W$.

Возьмем $v_j \in C^1( \Omega )$, которые сходятся к $u \in Lip$ вместе с производными почти всюду. Тогда, по теореме
Егорова, за исключение множества $A_\eps$: $\abs{ A_\eps } \le \eps$ они сходятся равномерно. Для каждой $v_j$
найдутся кусочно-линейные $w_{jk} \to v_j$ почти всюду вместе с производными. Аналогично, по теореме Егорова,
найдутся множества $B_{j\eps}$: $\abs{ B_{j\eps} } \le \frac{\eps}{2^j}$, вне которых сходимость равномерная.
Далее, для каждого $j$ возьмем $k_j$, начиная с которого вне $B_{j\eps}$ будет $\abs{ w_{jk_j} - v_j } < \frac{1}{2^j}$ и 
$\norm{ w_{jk_j}' - v_j' } < \frac{1}{2^j}$. Тогда вне множества $\cup_j B_{j\eps} \cup A_\eps$ $w_{jk_j} \to u$
равномерно вместе с производными. В силу малости этого множества и произвольности $\eps$, получаем, что
$w_{jk_j} \to u$ почти всюду с производными. Обозначим $u_j = w_{jk_j}$.

$F( x', u_j(x), \norm{ a_i(x',u_j(x))( u_j )'_i(x),a(x,u_j(x))( u_j )'_n(x)})$
ограничена как композиция непрерывных функций, ограниченных в совокупности.
Таким образом, $I( u_j )$ ограничены, подынтегральные выражения
сходятся почти всюду и можно применить теорему Лебега. Получаем, что $I( u_j ) \to I( u )$.

Тем самым мы попадаем в условия утверждения (\ref{uplift}) при
$A = W^1_\infty(-1,1)$ и $B$, содержащем все кусочно-линейные функции.
\end{proof}

\section{Переход к $\W$-функциям}

Поднимем результат на случай обобщенно дифференцируемых функций в одномерном случае: $\Omega = [-1,1]$.

\begin{lm} \cite[Lemma 2.7]{ASC}
Пусть $\phi_h: \Omega \to \Real$ --- последовательность липшицевых функций, удовлетворяющих условиям:
$\phi_h' \ge 1$ для почти всех $x$ и всех $h$, $\phi_h( x ) \to x$ для почти каждого $x$.
Тогда для любой $f \in L_1(\Real)$ $f(\phi_h) \to f$ в $L_1(\Real)$.
\end{lm}

\begin{prop}
\label{Wapprox}
Если функция $a$ непрерывна, четна и убывает на $[0, 1]$, то
$\forall u \in \W \exists u_n \in Lip: u_n \to u$ в $\W$ и $I(u_n) \to I(u)$.
\end{prop}

\begin{proof}
Можно считать, что $I( u ) < \infty$.

Вес $a$ возрастает по $x$ при $x \in \Omega_1 = [-1, 0]$
и убывает при $x \in \Omega_2 = [0, 1]$.
Докажем утверждение для функционала $$I_2( u ) = \int_{\Omega_2} F( u(x), a(x, u(x)) u'(x) ) dx$$
а случай $\Omega_1$ сведем к первому: $$I_1( u ) = \int_{\Omega_1} F( u( x ), a( x, u(x) ) u'( x ) ) dx =
\int_{-\Omega_1} F( u( -z ), a( -z, u(-z) ) u'( -z ) ) dz.$$

Известно, что для любой $u$ найдется последовательность $v_h \in Lip$ такая, что $v_h = u$ и $v_h' = u'$
на множествах $\Omega_2 \setminus A_h$, где $\abs{A_h} \to 0$, причем $A_h$ монотонно убывают по включению.
Очевидно, можно выбрать $v_h$ так, что они будут линейны на связных компонентах $A_h$ ($A_h$ --- открытые),
а также $v_h(0) = u( 0 )$.
Тогда $v_h \to u$ в $\W$. Кроме того,
$$I_2(v_h)=\int_{\Omega_2 \setminus A_h} F( u(x), a( x, u(x) ) u'( x ) ) dx +
\int_{A_h} F( v_h(x), a( x, v_h(x) ) v_h'( x ) ) dx.$$
Первое слагаемое сходится к $I_2(u)$ при $h \to \infty$.
Изменим немного аппроксимацию, чтобы доказать сходимость второго слагаемого к $0$.

Представим $A_h = \cup \Omega_{h,k}$, где $\Omega_{h,k} = ( b_{h,k}^-, b_{h,k}^+ )$. $\alpha_{h,k} = \abs{\Omega_{h,k}}$,
$\beta_{h,k} = v_h(b_{h,k}^+) - v_h(b_{h,k}^-) = u(b_{h,k}^+) - u(b_{h,k}^-)$.
Тогда $v'_h = \frac{ \beta_{h,k} }{ \alpha_{h,k} }$ в $\Omega_{h,k}$.
Заметим, что $\sum \abs{\beta_{h,k}} \le \int_{A_h} \abs{u'} dx < \infty$, а значит,
$\sum \abs{\beta_{h,k}} \to 0$.

Рассмотрим некоторую функцию $\phi_h \in \W(\Omega_2)$ такую, что
\begin{align*}
\phi_h( 0 ) &= 0 & & \\
\phi_h' &=  1 & \text{ в } & \Omega_2 \setminus A_h,\\
\phi_h' &=  \frac{ \abs{\beta_{h,k}} }{ \alpha_{h,k} } \vee 1 & \text{ в } & \Omega_{h,k},
\end{align*}
где $t_1 \vee t_2 = \max\{ t_1, t_2 \}$.

Заметим, что $\int_{\Omega_2} \abs{\phi_h'} dx \le 2 + \sum \abs{ \beta_{h,k} } < \infty$.

{\bf Факт 1: } $\phi_h \to Id$ в $\W(\Omega_2)$, где $Id( x ) = x$.

Достаточно доказать, что $\phi_h' \to 1$ в $L_1( \Omega_2 )$:
$$\int \abs{ \phi_h' - 1 } dx = \sum ( \frac{ \abs{ \beta_{h,k} } }{ \alpha_{h,k} } \vee 1 - 1 ) \alpha_{h,k} \le
\sum \abs{ \beta_{h,k} } \to 0$$.

{\bf Факт 2: } $\abs{\phi_h( A_h )} \to 0$

$$\abs{\phi_h( A_h )} = \sum \abs{ \phi_h( \Omega_{h,k} ) } = \sum \abs{\beta_{h,k}} \vee \alpha_{h,k}
\le \sum \alpha_{h,k} + \sum \abs{ \beta_{h,k} } \to 0$$.

Рассмотрим теперь $\phi_h^{-1}: \Omega_2 \to \Omega_2$ --- ограничение обратной к $\phi_h$ функции на $\Omega_2$.
Для нее верно $0 \le ( \phi_h^{-1} )' \le 1$ и

\begin{align*}
\phi_h^{-1} ( 0 ) &= 0 & & \\
( \phi_h^{-1} )' &=  1 & \text{ в } & \Omega_2 \setminus \phi_h( A_h ),\\
( \phi_h^{-1} )' &=  \frac{ \alpha_{h,k} }{ \abs{ \beta_{h,k} } } \wedge 1 & \text{ в } & \Omega_2 \cap \phi_h( \Omega_{h,k} ).
\end{align*}

Тогда возьмем $u_h = v_h( \phi_h^{-1} )$. Она липшицева как композиция липшицевых функций. Кроме того,

\begin{align*}
u_h( 0 ) &= u( 0 ) & & \\
u_h' &=  v_h'( \phi_h^{-1} ) \cdot ( \phi_h^{-1} )' = u'( \phi_h^{-1} ) & \text{ в } & \Omega_2 \setminus \phi_h( A_h ),\\
u_h' &=  v_h'( \phi_h^{-1} ) \cdot ( \phi_h^{-1} )' = 
\sign{ \beta_{h,k} } ( 1 \wedge \frac{ \abs{ \beta_{h,k} } }{ \alpha_{h,k} } ) & \text{ в } & \Omega_2 \cap \phi_h( \Omega_{h,k} ).
\end{align*}

{\bf Факт 3: } $u_h \to u$ в $\W( \Omega_2 )$.

Достаточно доказать $u_h' \to u'$ в $L_1( \Omega_2 )$:
$$\norm{ u_h' - u' }_{L_1} \le \int_{ \Omega_2 \setminus \phi_h(A_h) } \abs{ u_h' - u' } + 
\int_{ \Omega_2 \cap \phi_h(A_h) } \abs{ u_h'} + \int_{ \Omega_2 \cap \phi_h(A_h) } \abs{ u'} = P_h^1 + P_h^2 + P_h^3$$
$$P_h^1 = \int_{ \Omega_2 \setminus \phi_h( A_h ) } \abs{ u'( \phi_h^{-1} ) - u' } dx =
\int_{ \phi_h^{-1} ( \Omega_2 ) \setminus A_h } \abs{ u' - u'( \phi_h ) } dz \le
\int_{\Omega_2} \abs{ u' - u'( \phi_h ) } dz$$
Тогда, в силу леммы $P_h^1 \to 0$.

$P_h^2 \le \abs{ \phi_h( A_h ) } \to 0$

$P_h^3 \to 0$ по абсолютной непрерывности интеграла.

{\bf Факт 4: } $I_2( u_h ) \to I_2( u )$.

$$I_2( u_h ) = \int\limits_{ \Omega_2 \setminus \phi_h( A_h ) } F( u_h( x ), a( x, u_h(x) ) u_h'( x ) ) dx +\
\int\limits_{ \Omega_2 \cap \phi_h( A_h ) } F( u_h( x ), a( x, u_h(x) ) u_h'( x ) ) dx.$$
Обозначим эти слагаемые $\hat{P_h^1}$ и $\hat{P_h^2}$.
Так как $u \in \W( \Omega_2 )$, то $u \in L_\infty( \Omega_2 )$. Обозначим $\norm{ u }_\infty = r$,
тогда $\norm{ u_h }_\infty < 2r$ с некоторого $h$. Кроме того, $\abs{ u_h' } \le 1$
почти всюду в $\phi_h( A_h )$. Тогда $\hat{P_h^2} \le M_F \abs{ \phi_h( A_h ) } \to 0$, где
$M_F = \max F$ по множеству $[-2r, 2r] \times [-M_a, M_a]$, а $M_a = \max a$ по $\Omega_2 \times [-2r, 2r]$.

\begin{multline*}
\hat{P_h^1} = \int\limits_{ \Omega_2 \setminus \phi_h( A_h ) }
	F( u( \phi_h^{-1}( x ) ), a( x, u( \phi_h^{-1}( x ) ) u'( \phi_h^{-1}( x ) ) ( \phi_h^{-1} )' ) dx =
\\ =\int\limits_{ \phi_h^{-1}( \Omega_2 ) \setminus A_h } F( u( z ), a( \phi_h( z ), u( z ) ) u'( z ) ) dz
\\ = \int\limits_{ \Omega_2 } F( u( z ), a( \phi_h( z ), u( z ) ) u'( z ) ) \chi_{ \phi_h^{-1}( \Omega_2 ) \setminus A_h }dz
\end{multline*}
Последнее равенство, вообще говоря, не имеет смысла, так как $\phi_h( z )$ может принимать значения вне $\Omega_2$.
Определим $a( z, u ) = a( 1, u )$ при $z > 1$. Теперь выражение корректно.
Заметим, что множества $\phi_h^{-1}( \Omega_2 ) \setminus A_h$ возрастают, так как множества
$\phi_h^{-1}( \Omega_2 )$ возрастают и $A_h$ убывает, то есть
$\phi_{h_1}^{-1}( \Omega_2 ) \subset \phi_{h_2}^{-1}( \Omega_2 )$ и $A_{h_1} \supset A_{h_2}$ при $h_1 \le h_2$.
На отрезке $\Omega_2$ (и даже $\phi_h( \Omega_2 )$) $a$ убывает, значит $a( \phi_h( z ) )$ будет расти по $h$,
так как $\phi_h( z )$ убывает по $h$. В таком случае можно применить теорему
о монотонной сходимости и получить в пределе $$\hat{P_h^1} \to \int_{\Omega_2} F( u( z ), a( z, u( z ) ) u'( z ) ) dz.$$

\end{proof}

\rem Если функция $a$ четна, вогнута и неотрицательна, то она удовлетворяет условию (\ref{almostConcave}).
Действительно: для любых $s$ и $t$ $a( 1 ) - a( s ) \le a( t ) - a( -1 + t - s )$. А так как $a( 1 ) \ge 0$, то
получаем $a( s ) + a( t ) \ge a( -1 + t - s ) = a( 1 - t + s )$. Обратное вообще говоря неверно, то есть не всякая четная
неотрицательная функция, удовлетворяющая (\ref{almostConcave}) вогнута.

\rem Очевидно, что те же рассуждения с закреплением функции $u$ на левом конце можно провести на любом интервале по $x$, где
вес $a$ убывает по $x$. То есть получить на этом интервале последовательность $u_h \to u$ в $\W$ и $I( u_h ) \to I( u )$.
Аналогично, можно закрепить правый конец при условии возрастания $a$ по $x$.

\begin{prop}
Если функция $a$ непрерывна, четна и убывает на $[0, 1]$, то для любой $u \in \W$ $I( u^* ) \le I( u )$.
\end{prop}

\begin{proof}
Неравенство немедленно следует из утверждений (\ref{uplift}) и (\ref{Wapprox}).
\end{proof}

\section{Переход к $\W$-функциям в случае симметричной перестановки}
\label{sobolevSymm}

Как и раньше (в случае с переходом к соболевским функциям), нам нужно для каждой $u \in \W$ лишь найти
аппроксимацию липшицевыми $u_h$: $u_h \to u$ в $\W$ и $I( u_h ) \to I( u )$. Остальные неравенства обеспечиваются слабой
полунепрерывностью снизу фунционала $I$. Отличие состоит в том, что в случае с монотонной перестановкой можно было закрепить
$u$ в $0$ и аппроксимировать ее отдельно на $[-1,0]$ и $[0,1]$. Поскольку в случае симметричной перестановки вес $a$ имеет
противоположную монотонность, то закреплять нужно на концах интервала. Но при склейке аппроксимаций в $0$ не получится функция
из $\W$, так как склейка может иметь разрыв в $0$.
Для начала, найдем нужную аппроксимацию при условии, что найдется некоторое $t$, что $a( x, u )$ зависит только
от $u$ при $\abs{ x } \le t$.

Найдем точку $x_0 \in [-t, t]$, удовлетворяющую
\begin{equation}
\label{meanConv}
\frac{u(x_0 + \eps) - u(x_0 - \eps)}{2 \eps} \to u'(x_0).
\end{equation}
Такие точки всюду плотны на любом отрезке. Зафиксируем $x^-$ и $x^+$ такие, что $x^- + x^+ = 2 x_0$.
Определим $w( x ) = u( x )$ при $x \in [ -1, x^- ] \cup [ x^+, 1 ]$, а на отрезке $[ x^-, x^+ ]$ линейным образом.
Тогда $w \in \W$.
Если приближать функцию $u$ на интервале $[-1, x_0]$,
закрепив левый конец, получим последовательность липшицевых $u_h^-$. А также в процессе построения будут получены
множества замененных точек $A_h^-$, функции $v_h^-$, совпадающие с $u$ на $[-1, x^-] \setminus A_h^-$,
и $\phi_h^-$, соответствующие $A_h$, $v_h$ и $\phi_h$ из построения. На интервале $[x_0, 1]$,
закрепив правый конец, получим аналогично последовательности $u_h^+$, $A_h^+$, $v_h^+$, $\phi_h^+$.
Функции $u_h^-$ и $u_h^+$ приближают $w$ на отрезках $[ -1, x^- ]$ и $[ x^+, 1 ]$.
Построим аппроксимирующую последовательность для $w$ следующим образом:
$w_h( x ) = u_h^-( x )$ при $x \le x^-$, $w_h( x ) = u_h^+( x )$ при $x \ge x^+$ и
линейно при $x \in [x^-, x^+]$:
$w_h( x ) = u_h^-( x^- ) + ( u_h^+( x^+ ) - u_h^-( x^- ) ) \frac{ x - x^- }{ x^+ - x^- }$.

\begin{multline*}
\abs{ \int_{-1}^1 F( w, a( x, w ) \abs{ w' } )dx - \int_{-1}^1 F( w_h, a( x, w_h ) \abs{ w_h' } )dx } \le \\
\abs{ \int_{-1}^{x^-} ( F( w, a( x, w ) \abs{ w' } ) - F( w_h, a( x, w_h ) \abs{ w_h' } ) )dx } + \\
\abs{ \int_{x^+}^1 ( F( w, a( x, w ) \abs{ w' } ) - F( w_h, a( x, w_h ) \abs{ w_h' } ) )dx } + \\
\abs{ \int_{x^-}^{x^+} F( w, a( x, w ) \abs{ w' } ) - F( w_h, a( x, w_h ) \abs{ w_h' } )dx } =:
\bar{ P_1 } + \bar{ P_2 } + \bar{ P_3 }
\end{multline*}

$\bar{ P_1 },\bar{ P_2 } \to 0$ при $h \to \infty$ по построению. Если мы докажем, что $w_h( x^- ) \to w( x^- )$ и
$w_h( x^+ ) \to w( x^+ )$, то $w_h( x ) \to w( x )$ при $x \in [ x^-, x^+ ]$, то есть
$F( w_h, a( x, w_h ) \abs{ w_h' } ) \to F( w, a( x, w ) \abs{ w' } )$ поточечно.
Кроме того, на $[ x^-, x^+ ]$ с некоторого $h$ выполняется $\abs{ w_h'( x ) } < 2 \abs{ w'( x ) } + 1$
(обе функции линейны на этом участке)
и $\abs{ w_h( x ) } < 2 \abs{ w( x ) } + 1$. Тогда
$$\abs{ F( w, a( x, w ) \abs{ w' } ) - F( w_h, a( x, w_h ) \abs{ w_h' } ) } < 2 \max F( w, a( x, w ) p ) < \infty,$$
где максимум берется по множеству
$( x, w, p ) \in [ x^-, x^+ ] \times [ -2 \max \abs{ w } - 1, 2 \max \abs{ w } + 1 ] \times [ 0, 2 \abs{ w'( x_0 ) } + 1]$.
Тогда имеется интегрируемая мажоранта и по теореме Лебега $\bar{ P_3 } \to 0$.
Таким образом, осталось показать, что $w_h( x^- ) \to w( x^- )$ ($w_h( x^+ ) \to w( x^+ )$ доказывается аналогично).
$w_h( x^- ) = u_h^-( x^- ) = v_h^-( ( \phi_h^- )^{-1} ( x^- ) )$.
Множество точек, замененных в функции $u$ на левом промежутке, обозначено $A_h^-$. При переходе от $v_h^-$ к $u_h^-$ оно
переходит в $\phi_h^-( A_h^- )$. Таким образом, $( \phi_h^- )^{-1} ( x^- ) \in [ x^- - \abs{ \phi_h^-( A_h^- ) }, x^- ]$.
То есть $w_h( x^- ) \in v_h^-( [ x^- - \abs{ \phi_h^-( A_h^- ) }, x^- ] \cap [ -1, x_0 ] )$.
Для любого $x \notin A_h^-$ выполняется $v_h^-( x ) = u( x )$.
Если же $x \in A_h^-$, то $x$ лежит на одном из отрезков, на которых функция заменена на линейную.
Поскольку функция $u$ непрерывная, то на
этом отрезке есть точка с тем же значением. Но длина отрезка не превышает $\abs{ A_h^- }$, так что
$v_h^-( [ x^- - \abs{ \phi_h^-( A_h^- ) }, x^- ] \cap [ -1, x_0 ] ) \subset
u( [ x^- - \abs{ \phi_h^-( A_h^- ) } - \abs{ A_h^- }, x^- + \abs{ A_h^- } ] \cap [ -1, x_0 ] ) =: S_h^-$ (начиная с момента, когда
$x^- + \abs{ A_h^- } < x_0$). А значит, $w_h( x^- ) \in S_h^-$. Из непрерывности $u$ следует, что $S_h^-$ стягивается в точку.
Тем самым, $w_h( x^- ) \to u( x^- ) = w( x^- )$.

Мы приблизили функцию $w$ липшицевыми $w_h$ по норме так, что $I( w_h ) \to I( w )$. $w$ зависит от точек $x^-$ и $x^+$.
Устремим их к $x_0$, сохраняя соотношение $x^- + x^+ = 2 x_0$ и покажем,
что $I( w ) \to I( u )$ ($w \to u$ в $\W$ очевидно по теореме Лебега).
$w$ сходится к $u$ почти всюду (а именно, кроме точки $x_0$).
\begin{multline*}
f( x ) := \abs{ F( u, a( x, u ) \abs{ u' } ) - F( w, a( x, w ) \abs{ w' } ) } =\\
\abs{ F( u, a( x, u ) \abs{ u' } ) - F( w, a( x, w ) \abs{ w' } ) } \chi_{[ x^-, x^+]} \le\\
F( u, a( x, u ) \abs{ u' } ) \chi_{[ x^-, x^+]} + F( w, a( x, w ) \abs{ w' } ) \chi_{[ x^-, x^+]} =: g( x ).
\end{multline*}
Первое слагаемое, входящее в $g$, суммируемо, поскольку $I( u ) < \infty$. Покажем, что второе слагаемое равномерно ограничено.
Поскольку для $x_0$ выполняется (\ref{meanConv}), то $w'( x ) = \frac{ u( x^+ ) - u( x^- ) }{ x^+ - x^- } \to u'( x_0 )$ для
любого $x \in [ x^-, x^+ ]$.
При достаточно близких $x^-$ и $x^+$ будет $\abs{ w' } < 2 \abs{ u'( x_0 ) } + 1$.
Тогда $F( w, a( x, w ) \abs{ w' } ) \le \max F( w, a( x, w ) p )$, где максимум берется по множеству
$( x, w, p ) \in [ -1, 1 ] \times [ -\max \abs{ u }, \max \abs{ u } ] \times [ 0, 2 \abs{ u'( x_0 ) } + 1]$.
То есть нашлась суммируемая мажоранта $g$ для функции $f$ и по теореме Лебега получаем $I( w ) \to I( u )$.

Осталось избавиться от условия постоянства $a$ при $\abs{ x } \le t$. Возьмем вес общего вида $a$ и соответствующий ему
функционал $I( u ) = \int_{-1}^1 F( u, a( x, u ) u' )dx$. Определим веса $a_t( x, u ) = a( 0, u )$ при $\abs{ x } \le t$
и $a_t( x, u ) = a( x, u ) - a( t \sign{ x }, u ) + a( 0, u )$ при $\abs{ x } > t$. Очевидно, что $a_t \nearrow a$ при $t \to 0$,
непрерывны и выпуклы по $x$, то есть для них получена сходимость функционала.
Тогда для каждой из них мы имеем неравенство $F( \bar{u}, a_t( x, \bar{u} ) \abs{ \bar{u}' } ) \le F( u, a_t( x, u ) \abs{ u' } )$.
Но $F( u, a_t( x, u ) \abs{ u' } ) \nearrow F( u, a( x, u ) \abs{ u' } )$, а тогда, по теореме о монотонной сходимости,
$\int_{-1}^1 F( u, a_t( x, u ) u' )dx \to I( u )$ для произвольной $u$.

\rem Видно, что мы здесь не пользовались выпуклостью веса $a$, а лишь его монотонностью справа и слева от нуля. Однако, для получения
результата для симметричной перестановки (в частности, для получения неравенства на кусочно-линейных функциях)
выпуклость необходима.

\section{Избавление от требования кусочной монотонности $a$}

!!! Переписать для произвольных $a( x, u )$ !!!

Получим теперь неравенство для произвольных функций $a( x, u )$ вида
$a^0( x ) \varkappa( u )$, но для пространства
функций $u \in W(-1, 1)$, постоянных в $\eps$-окрестности $\pm 1$.

Зафиксируем некоторое $\eps$ и некоторую функцию $u \in \W(-1, 1)$, постоянную
на $[-1,-1 + \eps] \cup [1 - \eps, 1]$. Предположим, что
$\int_{-1}^1 F(u(x), a(x, u) u'(x))dx < \infty$. Возьмем число $\theta < 1$ и
число $k(\theta) \in \Nat$ так, чтобы $\theta a_k^0(x) \le a^0(x)$ при
$x \in [-1 + \eps, 1 - \eps]$. Здесь $a_k^0$ --- кусочно-линейное приближение
$a^0$, состоящее из $k$ отрезков, которое, в соответствии с леммой
(\ref{piecewiseLinearConcave}), удовлетворяет неравенству
(\ref{almostConcaveSum}). Покажем, что для любого
$\theta$ найдется такое $k$. Для любого $i$ будет выполнено
$\theta a_k^0(-1 + \frac{2i}{k}) < a^0(-1 + \frac{2i}{k})$. Для любого
неотрицательного $z < \frac{2}{k}$ по неравенству (\ref{almostConcaveSum})
$a^0(-1 + z) + a^0(1 - \frac{2i}{k} - z) \ge a^0(1 - \frac{2i}{k})$, то есть
$a^0(-1 + \frac{2i}{k} + z) \ge a^0(-1 + \frac{2i}{k}) - a^0(-1 + z)$, так как
$a^0$ четная. С другой стороны, по неравенству (\ref{almostConcaveSum})
$a_k^0(-1 + \frac{2i}{k} + z) \le a_k^0(-1 + \frac{2i}{k}) + a_k^0(-1 + z)$.
Тем самым, $a^0(-1 + \frac{2i}{k} + z) - \theta a_k^0(-1 + \frac{2i}{k} + z)
\ge (1 - \theta) a^0(-1 + \frac{2i}{k}) - a^0(-1 + z) - \theta a_k^0(-1 + z)$.
Но $a^0(-1 + z) + \theta a_k^0(-1 + z) \le
2 \max \{ a^0(-1 + \zeta): \zeta \in [0, \frac{2}{k}]\} \to 0$.
Оценим снизу $a^0(-1 + \frac{2i}{k})$. Поскольку нас интересует неравенство на
$[-1 + \eps, 1 - \eps]$, нам достаточно оценки для
$-1 + \frac{2i}{k} \in [-1 + \eps - \frac{2}{k}, 1 - \eps + \frac{2}{k}]$.
А начиная с некоторого $k$, достаточно получить оценку при
$-1 + \frac{2i}{k} \in [-1 + \frac{\eps}{2}, 1 - \frac{\eps}{2}]$.
$a^0(-1 + \frac{2i}{k}) \ge
\min \{ a^0(x): x \in [-1 + \frac{\eps}{2}, 1 - \frac{\eps}{2}] \}$,
что положительно и не зависит от $k$.
Значит, найдется $k(\eps, \theta)$ такое, что $\theta a_k^0(x) \le a^0(x)$ при
$x \in [-1 + \eps, 1 - \eps]$. Положим $\theta_j = 1 - \frac{1}{j}$,
$b_j(x) = \max\limits_{i \le j} \theta_i a_{k(\theta_i)}^0(x)$ и
$a_j(x, u) = \varkappa(u) b_j(x)$. По лемме (\ref{maxConcave}) $b_j$
удовлетворяют неравенству (\ref{almostConcaveSum}).
$a_j(x, u) \le a(x, u)$ при $x \in [-1 + \eps, 1 - \eps]$ и
$a_j(x, u) \nearrow a(x, u)$.

Функции $b_j$ кусочно-монотонные как максимум нескольких кусочно-монотонных
функций, поэтому
\begin{equation}
\label{inequlityForPartiallyLinear}
\int_{-1}^1 F(u(x), a_j(x, u) u'(x)) dx \ge
\int_{-1+\eps}^{1-\eps} F(u^*(x), a_j(x, u) {u^*}'(x)) dx.
\end{equation}
Покажем, что 
\begin{align*}
\int_{-1}^1 F(u(x), a_j(x, u) u'(x)) dx \to &
\int_{-1}^1 F(u(x), a(x, u) u'(x)) dx \\
\int_{-1+\eps}^{1-\eps} F(u^*(x), a_j(x, u) {u^*}'(x)) dx \to &
\int_{-1+\eps}^{1-\eps} F(u^*(x), a(x, u) {u^*}'(x)) dx.
\end{align*}
В обоих случаях подынтегральная функция левой части интегрируема,
монотонно возрастает по $j$ и сходится к подынтегральной функции правой части.
Тем самым, по теореме о монотонной сходимости получаем
$$I(u) = \int_{-1}^1 F(u(x), a(x, u) u'(x)) dx \ge
\int_{-1+\eps}^{1-\eps} F(u^*(x), a(x, u) {u^*}'(x)) dx.$$

Возьмем теперь произвольную $u \in \W(-1,1)$ и $u_\eps(x) = u(x)$ при
$x \in [-1 + \eps, 1 - \eps]$ и постоянную при остальных $x$. Тогда
$u_\eps \to u$ в $\W(-1,1)$ и $I(u_\eps) \to I(u)$. Поясним последнее.
$u_\eps \to u$ поточечно. Кроме того,
$I(u_\eps)=\eps F(u(-1 + \eps), 0) +
\int_{-1+\eps}^{1 - \eps} F(u(x), a(x) u'(x))dx + \eps F(u(1 - \eps), 0) \le
I(u) + 2\eps \max\{F(u(x), 0): x \in [-1,1]\} < \infty$. То есть, по теореме о
мажорируемой сходимости $I(u_\eps) \to I(u)$.
Поскольку $u_\eps \to u$ в $\W(-1+\eps,1-\eps)$,
$u_\eps \rightharpoondown u$ в $\W(-1+\eps,1-\eps)$. Значит
$\int_{-1+\eps}^{1-\eps} F(u^*(x), a(x, u) {u^*}'(x)) dx \le
\varliminf
\int_{-1+\eps}^{1-\eps} F(u_\eps^*(x), a(x, u_\eps) {u_\eps^*}'(x)) dx$.
Тем самым,
$$I(u) \ge \int_{-1+\eps}^{1-\eps} F(u^*(x), a(x, u) {u^*}'(x)) dx$$
уже для произвольно функции $u \in \W(-1,1)$. Но $\eps$ здесь можно взять
любое, поэтому получаем

\begin{prop}
Если функция $a(x,u)$ представима в виде $a^0(x) \varkappa(u)$, где
$a^0$ удовлетворяет условию (\ref{almostConcaveSum}), то для любой
$u \in \W(-1,1)$ $I(u^*) \le I(u)$.
\end{prop}

На протяжении всей работы мы накладываем на вес $a$ следующие ограничения:

\smallskip
\noindent

Будем называть такие функции допустимыми.

Далее будем накладывать по одному новые условия на вес $a$, каждый раз
показывая, что от нового условия можно избавиться, с тем, чтобы войти в
условия, при которых результат уже доказан.

Схема: $a$ удовлетворяет $H1$--$Hi$.
Строим последовательность $a_k$, удовлетворяющих $H1$--$H(i+1)$
так, чтобы $a^n := \max\limits_{k \le n} a_k$ тоже удовлетворяли
$H1$--$H(i+1)$ и выполнялось
\begin{align*}
F(u(x), a^n(x, u(x)) u'(x)) & \nearrow F(u(x), a(x, u(x)) u'(x))\\
F(u^*(x), a^n(x, u^*(x)) {u^*}'(x)) & \nearrow F(u^*(x), a(x, u^*(x)) {u^*}'(x))
\end{align*}
для почти всех $x$.

\begin{prop}
\label{levelDerivative}
\cite[стр. 137]{LL}
Для любой $u \in \W(-1, 1)$ и произвольного $u_0 \in \Real$ выполнено
$u'(x) = 0$ для почти всех $x \in \{ x : u(x) = u_0 \}$.
\end{prop}

\begin{lm}
\label{zeroApprox}
Предположим, вес $a$ является допустимым и имеется замкнутое множество
$W \subset \Real$ такое, что либо оно содержит счетное число элементов,
либо $a|_{u \in W} \equiv 0$.
Тогда найдется последовательность весов $b_k$ такая, что $I(b_k, u) \to I(a, u)$,
и для любого $u$ в некоторой окрестности $W$ (зависящей от $k$) $b_k(\cdot, u) \equiv 0$,
а также $b_k \nearrow a$ для всех $u$, кроме, возможно, счетного количества.
\end{lm}

\begin{proof}
Возьмем $b_k(x, u) := a(x, u) \min(1, \max(0, k \dist(u, W) - 1)) \le a(x, u)$.
Этот вес равен нулю в $\left(\frac{1}{k}\right)$-окрестности $W$.
В частности, если $W$ не состоит из счетного числа элементов,
$b_k \equiv a \equiv 0$ при $u \in W$.
Кроме того, $b_k \equiv a$ вне $\left(\frac{2}{k}\right)$-окрестности $W$ и
$b_k(x, u)$ возрастают при увеличении $k$.
Тем самым, $b_k \nearrow a$ при всех $u$ кроме, возможно, счетного количества точек.
Отсюда $I(\Real \setminus W, b_k, u) \nearrow I(\Real \setminus W, a, u)$.

Если $W$ не состоит из счетного числа точек,
$I(W, b_k, u) = \int_W F(u(x), 0) dx = I(W, a, u)$.
В противном случае
$W = \{w_i\}_{i = 1}^\infty$ и
$$I(W, b_k, u) = \sum_{i = 1}^\infty \int\limits_{x : u(x) = w_i} F(w_i, b_k(x, w_i) u'(x)) dx.$$
При этом, по утверждению (\ref{levelDerivative}), на каждом из множеств
$\{ x : u(x) = w_i \}$ производная $u'(x)$ почти всюду равна нулю.
То есть,
$I(W, b_k, u) = \sum_{i = 1}^\infty \int\limits_{x : u(x) = w_i} F(w_i, 0) dx = I(W, a, u)$
(такой интеграл не зависит от веса).
\end{proof}

Применим лемму \ref{zeroApprox}. В качестве множества $W$ возьмем множество
всех $u$, при которых происходит переход графика $u(x)$ из прямоугольника,
в котором вес убывает по $x$ в прямоугольник, в котором вес возрастает.
Очевидно, получившиеся $b_k$ удовлетворяют $(H7)$.

\section{Дополнение. Необходимые условия в случае симметричной перестановки}

Будем рассматривать условия, необходимые для выполнения
неравенства $I(\overline{u}) \le I(u)$ в одномерном случае. Как и для монотонной перестановки, можно
считать, что $F$ зависит только от последнего аргумента, а $a$ --- только от первого.

\begin{prop}
$a$ должна удовлетворять условию $a( s ) + a( t ) \ge a( \frac{ s - t }{2} ) + a( \frac{ t - s }{2} )$ для
любых $s$ и $t$.
\end{prop}

\begin{proof}
Пусть, для определенности, $s < t$. И пусть
$a( s + z ) + a( t - z ) + 2 \delta < a( \frac{ s - t + 2 z }{2} ) + a( \frac{ t - s - 2 z }{2} )$
для $0 \le z \le \eps$, где $2 \eps < t - s$.
Построим функцию $u$: $u( x ) = 0$, если $x \le s$ или $x \ge t$; $u( x ) = 1$ при $s + \eps \le x \le t - \eps$; на
оставшихся промежутках она будет линейная с производной $\frac{1}{\eps}$ и $-\frac{1}{\eps}$ соответственно.
Тогда $\overline{u}( x ) = 0$ при $x \le \frac{ s - t }{2}$ и при $x \ge \frac{t - s}{2}$;
$\overline{u}( x ) = 1$ при $\frac{s - t}{2} + \eps \le x \le \frac{t - s}{2} - \eps$; на оставшихся двух кусках
опять линейная с теми же производными.

Тогда 
\begin{multline*}
0 \le I(u) - I(\overline{u}) =
\\ \int_s^{s + \eps} F( \frac{a( x )}{\eps} ) dx + \int_{t - \eps}^{t} F( \frac{a( x )}{\eps} ) dx -
\int_{\frac{s - t}{2}}^{\frac{s - t}{2} + \eps} F( \frac{a( x )}{\eps} ) dx -
\int_{\frac{t - s}{2} - \eps}^{\frac{t - s}{2}} F( \frac{a( x )}{\eps} ) dx =: J.
\end{multline*}
Обозначим $A = \max a$. Подберем $F( x ) = x^\alpha$ так,
чтобы $\alpha > 1$ и $x \le F( x ) \le x + \delta$ для $x \le \frac{ A }{\eps}$.
Для этого возьмем $\alpha = \log_{ \frac{A}{\eps} }(\delta + \frac{A}{\eps})$.
Тогда $$J \le \int_s^{s + \eps} \frac{a( x )}{\eps} dx + \int_{t - \eps}^{t} \frac{a( x )}{\eps} dx -
\int_{\frac{s - t}{2}}^{\frac{s - t}{2} + \eps} \frac{a( x )}{\eps} dx -
\int_{\frac{t - s}{2} - \eps}^{\frac{t - s}{2}} \frac{a( x )}{\eps} dx < 0.$$

Для $s = t$ неравенство верно по непрерывности.
\end{proof}

\begin{prop}
Если функция $a \in C^1([-1, 1])$ удовлетворяет $a( s ) + a( t ) \ge a( \frac{ s - t }{2} ) + a( \frac{ t - s }{2} )$ для
любых $s$ и $t$, то она четная.
\end{prop}

\begin{proof}
Рассмотрим функцию $b( t ) = a( s ) + a( t ) - a( \frac{ s - t }{2} ) - a( \frac{ t - s }{2} ) \ge 0$.
При $t = -s$ $b( t ) = 0$, то есть достигает минимума. Значит ее производная 
$b'( -s ) = a'( -s ) + \frac{1}{2} a'( s ) - \frac{1}{2} a'( -s ) = 0$, то есть $a'( s ) = -a'( -s )$.
Тем самым, функция четна.
\end{proof}

\begin{prop}
Если функция $a \in C([-1, 1])$ такова, что $a( s ) + a( t ) \ge a( \frac{ s - t }{2} ) + a( \frac{ t - s }{2} )$ для
любых $s$ и $t$, то она четная.
\end{prop}

\begin{proof}
Продолжим $a( x ) = a( -1 )$ при $x < -1$ и $a( x ) = a( 1 )$ при $x > 1$.
Рассмотрим усреднение функции: $a_\rho( x ) = \int_\Real w_\rho ( z ) a( x - z ) dz = \int_\Real w_\rho ( z ) a( x + z ) dz$.
Тогда 
\begin{multline*}
a_\rho( s ) + a_\rho( t ) - a_\rho( \frac{ s - t }{2} ) - a_\rho( \frac{ t - s }{2} ) =
\\ \int_\Real w_\rho ( z ) ( a( s - z ) + a( t + z ) - a( \frac{ s - t }{2} - z ) - a( \frac{ t - s }{2} + z ) ) dz \ge 0.
\end{multline*}
Значит $a_\rho$ --- четная. Кроме того, $a_\rho( x ) \to a( x )$ равномерно. Тогда
$a( x ) = \lim a_\rho( x ) = \lim a_\rho( -x ) = a( -x )$.
\end{proof}

\begin{prop}
Если четная функция $a$ удовлетворяет $a( s ) + a( t ) \ge a( \frac{ s - t }{2} ) + a( \frac{ t - s }{2} )$ для
любых $s$ и $t$, то она выпукла.
\end{prop}

\begin{proof}
Действительно, для любых $s$, $t$ $a( s ) + a( t ) = a( s ) + a( -t ) \ge 2 a( \frac{ s + t }{2} )$.
\end{proof}
