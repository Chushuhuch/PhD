\section{Введение}

Положим $u \in \Wf[-1, 1]$, $u \ge 0$.
Обозначим через $u^*$ симметричную перестановку функции $u$.
Тогда выполнено классическое неравенство Пойя-Сеге:

\begin{equation}
\label{PS}
\mathcal{I}(u^*) \le \mathcal{I}(u), \qquad \text{где } \mathcal{I}(u) = \int\limits_{-1}^1 |u'(x)|^p dx, \quad p \ge 1
\end{equation}

Обобщения неравенства (\ref{PS}) обсуждаются в большом количестве работ (см. \cite{Tal} и ссылки оттуда).
В статье \cite{Br} показано, что неравенство (\ref{PS}) выполняется для функционалов вида
$$\int_{-1}^1 F( u(x), a( x, u( x ) ) \abs{ u'( x ) } ) dx,$$
где $a$ выпукла, а $F$ непрерывна и выпукла по второму аргументу, а также для многомерного аналога.
В статье \cite{1dim} результат статьи \cite{Br} доказан для естественного класса функций $u$.
Кроме того, доказано аналогичное неравенство для монотонной перестановки.

В этой работе мы вводим в функционал зависимость степени $p$ от $x$:
\begin{equation}
J(u) = \int\limits_{-1}^1 |u'(x)|^{p(x)} dx, \qquad I(u) = \int\limits_{-1}^1 ( 1 + | u'(x) |^2 )^{p(x) \over 2} dx.
\end{equation}
Здесь $p(x) \ge 1$ --- непрерывная функция на $[-1, 1]$, $u \in \Wf[-1, 1]$.

Переменный показатель суммирования возникает, например, в процессе изучения вырожденных эллиптических уравнений (см. напр. \cite{Kuf}, \cite{Zh2}).


\section{Необходимые условия}

\begin{prop}
\label{uniform}
Предположим, неравенство $J(u^*) \le J(u)$ выполнено для любой кусочно линейной функции $u$.
Тогда $p(x) \equiv const$.
\end{prop}
\begin{proof}
Зафиксируем произвольное $x_0 \in (-1, 1)$.
Для каждого $\alpha > 0$ и $\eps > 0$ такого, что $[x_0 - \eps, x_0 + \eps] \subset [-1, 1]$,
определим функцию
$$
\left\{
\begin{aligned}
u_{\alpha,\eps}(x) &= 0, & x & \in [-1, x_0 - \eps] \cup [x_0 + \eps, 1] \\
u_{\alpha,\eps}(x) &= \alpha \cdot (x - (x_0 - \eps)), & x & \in [x_0 - \eps, x_0] \\
u_{\alpha,\eps}(x) &= -\alpha \cdot ((x_0 + \eps) - x), & x & \in [x_0, x_0 + \eps].
\end{aligned}
\right.
$$

Тогда
$$
\left\{
\begin{aligned}
u_{\alpha,\eps}^*(x) &= 0, & x & \in [-1, -\eps] \cup [\eps, 1] \\
u_{\alpha,\eps}^*(x) &= \alpha \cdot (x + \eps), & x & \in [-\eps, 0] \\
u_{\alpha,\eps}^*(x) &= -\alpha \cdot (x - \eps), & x & \in [0, \eps].
\end{aligned}
\right.
$$

Кроме того, почти всюду
\begin{eqnarray*}
\begin{aligned}
& |u_{\alpha,\eps}'(x)|     = 0,      & x & \in [-1, x_0 - \eps] \cup [x_0 + \eps, 1] \\
& |u_{\alpha,\eps}'(x)|     = \alpha, & x & \in [x_0 - \eps, x_0 + \eps] \\
& |u_{\alpha,\eps}^*{}'(x)| = 0,      & x & \in [-1, -\eps] \cup [\eps, 1] \\
& |u_{\alpha,\eps}^*{}'(x)| = \alpha, & x & \in [-\eps, \eps] \\
\end{aligned}
\end{eqnarray*}

Тогда
$$
J(u_{\alpha, \eps}) = \int\limits_{x_0 - \eps}^{x_0 + \eps} \alpha^{p(x)} dx, \qquad
J(u_{\alpha, \eps}^*) = \int\limits_{-\eps}^{\eps} \alpha^{p(x)} dx.
$$

Перейдя к пределу при $\eps \to 0$ в неравенстве
$$
{J(u_{\alpha, \eps}^*) \over 2 \eps} \le {J(u_{\alpha, \eps}) \over 2 \eps},
$$
получим $\alpha^{p(0)} \le \alpha^{p(x_0)}$, в силу непрерывности $\alpha^{p(x)}$.
Заметим, что при $\alpha > 1$ и $\alpha < 1$ это дает неравенства $p(0) \le p(x_0)$ и $p(0) \ge p(x_0)$ соответственно.
Тем самым, в силу непрерывности $p$, $p(x) \equiv const$.
\end{proof}

Заметим, что ввиду предложения \ref{uniform} дальнейшее изучение неравенства $J(u^*) \le J(u)$ теряет интерес.

\begin{prop}
Предположим, неравенство $I(u^*) \le I(u)$ выполнено для любой кусочно линейной функции $u$.
Тогда функция $p$ четна и выпукла, а также выпукла функция двух переменных
$$K(s, x) = s ( 1 + { 1 \over s^2 } )^{p(x) \over 2} \qquad s > 0,\ x \in [-1, 1].$$
\end{prop}

Нам потребуется
\begin{lm}
\label{convLm}
{\rm \cite[лемма 10]{1dim}.}
Предположим, функция $p$ непрерывна на $[-1, 1]$, и для нее выполнено
$$
\forall s, t \in [-1, 1] \qquad p(s) + p(t) \ge p({s - t \over 2}) + p({t - s \over 2}).
$$
Тогда функция $p$ четна и выпукла.
\end{lm}

\begin{proof}
Зафиксируем две точки на отрезке $-1 < x_1 < x_2 < 1$
и рассмотрим финитную кусочно линейную функцию с ненулевыми производными только в окрестностях $x_1$ и $x_2$.
А именно, для достаточно малого $\eps$:
\begin{equation}
\left\{
\begin{aligned}
u(x) &= 0, & x &\in [-1, x_1 - s \eps] \cup [x_2 + t \eps, 1]\\
u(x) &= 2 \eps, & x &\in [x_1 + s \eps, x_2 - t \eps]\\
u(x) &= \eps + { x - x_1 \over s }, & x &\in [x_1 - s \eps, x_1 + s \eps]\\
u(x) &= \eps - { x - x_2 \over s }, & x &\in [x_2 - t \eps, x_2 + t \eps]
\end{aligned}
\right.
\end{equation}

Тогда, в силу сделанных предположений, верно следующее:
\begin{multline*}
\int\limits_{ x_1 - s \eps }^{ x_1 + s \eps } ( 1 + { 1 \over s^2 } )^{p(x) \over 2} dx
+ \int\limits_{ x_2 - t \eps }^{ x_2 + t \eps } ( 1 + { 1 \over t^2 } )^{p(x) \over 2} dx \ge
\\ \ge \int\limits_{ { x_2 - x_1 \over 2 } - { s + t \over 2 } \eps }^{ { x_2 - x_1 \over 2 } + { s + t \over 2 } \eps } ( 1 + \frac{1}{ ( { s + t \over 2 } )^2 } )^{p(x) \over 2} dx
     + \int\limits_{ { x_1 - x_2 \over 2 } - { s + t \over 2 } \eps }^{ { x_1 - x_2 \over 2 } + { s + t \over 2 } \eps } ( 1 + \frac{1}{ ( { s + t \over 2 } )^2 } )^{p(x) \over 2} dx
\end{multline*}

Или, переходя к $\eps \to 0$,
\begin{equation}
\label{preConv}
s ( 1 + { 1 \over s^2 } )^{p(x_1) \over 2} + t ( 1 + { 1 \over t^2 } )^{p(x_2) \over 2}
\ge { s + t \over 2 } ( 1 + \frac{1}{ ( { s + t \over 2 } )^2 } )^{p( { x_2 - x_1 \over 2 } ) \over 2}
  + { s + t \over 2 } ( 1 + \frac{1}{ ( { s + t \over 2 } )^2 } )^{p( { x_1 - x_2 \over 2 } ) \over 2}.
\end{equation}

Положим в неравенстве (\ref{preConv}) $s = t$:
$$( 1 + { 1 \over s^2 } )^{p(x_1) \over 2} + ( 1 + { 1 \over s^2 } )^{p(x_2) \over 2}
\ge ( 1 + { 1 \over s^2 } )^{p( { x_2 - x_1 \over 2 } ) \over 2} + ( 1 + { 1 \over s^2 } )^{p( { x_1 - x_2 \over 2 } ) \over 2}.$$

Обозначив $\sigma = {1 \over s^2}$ и воспользовавшись разложением в ряд Тейлора по $\sigma$ в точке $\sigma = 0$, получим
$$\sigma p(x_1) + \sigma p(x_2) \ge \sigma p( { x_2 - x_1 \over 2 } ) + \sigma p( { x_1 - x_2 \over 2 } ) + r(\sigma),$$

где $r(s) = o(\sigma)$ при $\sigma \to 0$. То есть, для любых $x_1, x_2 \in [-1, 1]$ выполнено
$$p(x_1) + p(x_2) \ge p( { x_2 - x_1 \over 2 } ) + p( { x_1 - x_2 \over 2 } ).$$

Воспользовавшись леммой \ref{convLm}, получаем, что $p$ четна и выпукла.
Тогда, подставив в (\ref{preConv}) $x_1$ и $-x_2$ и воспользовавшись четностью $p$, получаем
$K(s, x_1) + K(t, x_2) \ge 2 K( { s + t \over 2 }, { x_1 + x_2 \over 2 } )$.
\end{proof}

\section{Доказательство неравенства}

\begin{lm}
\label{quasiConv}
Положим, $m$ --- четное положительное число, $s_k > 0$ ($k = 1 \dots m$), $-1 \le x_1 \le \dots \le x_m \le 1$.
Тогда, если $K(s, x)$ четна по $x$ и выпукла по совокупности аргументов, выполнено
$$\sum\limits_{k = 1}^{m} K(s_k, x_k) \ge
2 K({1 \over 2} \sum\limits_{k = 1}^{m} s_k, {1 \over 2} \sum\limits_{k = 1}^{m} (-1)^k x_k).$$
\end{lm}

\begin{proof}
Докажем сперва неравенство в случае $m = 2$.
В этом случае оно принимает следующий вид:
$$K(s_1, x_1) + K(s_2, x_2) \ge 2 K({s_1 + s_2 \over 2}, {x_2 - x_1 \over 2}).$$
Это неравенство немедленно следует из условий леммы.

Теперь докажем неравенство для произвольного четного $m > 2$.
Введем новую функцию $L(s, x) = K(s, x) - s = s \big( (1 + {1 \over s^2})^{p(x) \over 2} - 1 \big)$.
Заметим, что утверждение леммы равносильно аналогичному неравенству для $L$:
$$\sum\limits_{k = 1}^{m} L(s_k, x_k) \ge
2 L({1 \over 2} \sum\limits_{k = 1}^{m} s_k, {1 \over 2} \sum\limits_{k = 1}^{m} (-1)^k x_k).$$
В частности, это неравенство уже доказано для $m = 2$.
Также заметим, что функция $L$ убывает по $s$.
Это следует из того, что $L$ выпукла, а также
$$L_s(s, x) = (1 + {1 \over s^2})^{{p(x) \over 2} - 1} ( 1 - {p(x) - 1 \over s^2} ) - 1 \xrightarrow[s \to \infty]{} 0.$$
Тогда

\begin{multline*}
\sum\limits_{k = 1}^{m} L(s_k, x_k)
\ge L(s_1, x_1) + L(s_m, x_m)
\overset{a}{\ge} 2 L({s_1 + s_m \over 2}, {x_m - x_1 \over 2}) \ge \\
\overset{b}{\ge} 2 L({1 \over 2} \sum\limits_{k = 1}^{m} s_k, {x_m - x_1 \over 2})
\overset{c}{\ge} 2 L({1 \over 2} \sum\limits_{k = 1}^{m} s_k, {1 \over 2} \sum\limits_{k = 1}^{m} (-1)^k x_k).
\end{multline*}

Здесь в переходе (a) мы воспользовались выпуклостью $L$,
в переходе (b) убыванием $L(s, x)$ по $s$,
а в переходе (c) --- возрастанием $L(s, x)$ по $x$ при $x \ge 0$.

\end{proof}

\begin{lm}
Если функция $K(s, x)$ четна по $x$ и выпукла по совокупности аргументов,
то для любой кусочно линейной функции $u \in \Wf[-1, 1]$ выполнено неравенство $I(u^*) \le I(u)$.
\end{lm}

\begin{proof}
Положим $L \subset [-1, 1]$ --- множество точек перелома (включая концы отрезка) функции $u$.
Возьмем $U = u([-1, 1]) \setminus u(L)$, образ функции $u$ без образов точек излома.
Это множество и представляется в виде объединения конечного числа непересекающихся интервалов $U = \cup_j U_j$.
Заметим, что для каждого $j$ множество $u^{-1}(U_j)$ разбивается на четное число интервалов,
на каждом из которых функция $u$ совпадает с некоторой линейной функцией $y^j_k$, $k = 1, \dots, m_j$.
Для удобства считаем, что носители $y^j_k$ для каждого $j$ идут по порядку,
то есть $\sup dom(y^j_k) \le \inf dom(y^j_{k + 1})$.
Обозначим $b^j_k = |y^j_k{}'(x)|$.
Также заметим, что множества точек $x \in [-1, 1]$, на которых $u$ и $u^*$ имеют нулевую производную, имеют одинаковую меру $Z$.
Тогда
\begin{multline}
I(u) - Z = \sum\limits_j \int\limits_{u^{-1}(U_j)} (1 + u'^2(x))^{p(x) \over 2} dx
= \sum\limits_j \sum\limits_k \int\limits_{dom(y^j_k)} (1 + {y^j_k}'^2(x))^{p(x) \over 2} dx =
\\ = \sum\limits_j \int\limits_{U_j} \sum\limits_k {1 \over b^j_k} (1 + b^j_k{}^2)^{p((y^j_k)^{-1}(y)) \over 2} dy
\end{multline}

Поскольку функция $u^*$ четна и монотонна с любой стороны от нуля, для удобства обозначим за $(u^*)^{-1}$
обратную к функции $u^*$, ограниченной на отрезок $[0, 1]$.
В тех точках, где обратная функция не определена, возьмем любое значение.
Все дальнейшие соотношения на значения функций и их производных верны почти всюду.
Тогда для каждого $j$ можно, используя функции $y^j_k$, выразить функцию $(u^*)^{-1}$ на участке $U_j$ следующим образом
(здесь $y \in U_j$):
\begin{eqnarray*}
(u^*)^{-1} (y) & = & {1 \over 2} \sum\limits_{k = 1}^{m_j} (-1)^k (y^j_k)^{-1}(y) \\
|((u^*)^{-1})'(y)| & = & {1 \over |u^*{}'((u^*)^{-1}(y))|} = {1 \over 2} \sum\limits_{k = 1}^{m_j} {1 \over b^j_k} =: { 1 \over b_j^* }.
\end{eqnarray*}
Тогда
\begin{multline}
I(u^*) = 2 \int\limits_0^1 (1 + u^*{}'^2(x))^{p(x)} dx
= Z + 2 \int\limits_U |((u^*)^{-1})'(y)| ( 1 + { 1 \over ((u^*)^{-1})'(y)^2 } ) dy =
\\ = Z + 2 \sum\limits_j \int\limits_{U_j} {1 \over b_j^*}
\Big(1 + b_j^*{}^2 \Big)^{p\big({1 \over 2} \sum\limits_{k = 1}^{m_j} (-1)^k (y^j_k)^{-1}(y)\big)} dy
\end{multline}

Зафиксируем $j$ и $y$.
Тогда для доказательства леммы достаточно выполнения
$$\sum\limits_{k = 1}^{m_j} K({1 \over b^j_k}, (y^j_k)^{-1}(y)) \ge
2 K({1 \over 2} \sum\limits_{k = 1}^{m_j} {1 \over b^j_k}, {1 \over 2} \sum\limits_{k = 1}^{m_j} (-1)^k (y^j_k)^{-1}(y)).$$

Но это неравенство обеспечивается леммой \ref{quasiConv}
\end{proof}

Для снятия условия кусочной линейности с функций $u$ нам понадобится следующая

\begin{lm}[\cite{Zh1}, Теорема 2.2]
\label{regAppr}
Пусть модуль непрерывности $\omega(\eps)$ функции $p$ удовлетворяет условию
\begin{equation}
\label{logCont}
- \lim\limits_{\eps \to 0} \omega(\eps) \log \eps < \infty.
\end{equation}
Тогда для любой $u \in \Wf[-1, 1]$ такой, что $I(u) < \infty$,
найдется последовательность кусочно линейных функций $u_n$, определенных на $[-1, 1]$, для которых
$u_n \to u$ в $\Wf[-1, 1]$ и $I(u_n) \to I(u)$.
\end{lm}

\begin{lm}
\label{logContIneq}
Пусть $p$ четна и удовлетворяет условию (\ref{logCont}), а $K$ выпукла.
Тогда для любой функции $u \in \Wf[-1, 1]$ выполнено $I(u^*) \le I(u)$.
\end{lm}

\begin{proof}
Зафиксируем некоторую функцию $u \in \Wf[-1, 1]$.
Тогда, в силу леммы \ref{regAppr} найдется последовательность кусочно линейных функций $u_n \to u$ в $\Wf[-1, 1]$ таких, что $I(u_n) \to I(u)$.

В статье \cite{Br} доказано (Теорема 1), что из $u_n \to u$ в $\Wf[-1, 1]$ следует $u_n^* \rightharpoondown u^*$ в $\Wf[-1, 1]$.
Кроме того, поскольку $p$ непрерывна, она ограничена, тем самым функционал $I$ слабо полунепрерывен снизу (\comment{ссылку}).

Тогда, перейдя к нижнему пределу в неравенстве $I(u_n^*) \le I(u_n)$, получаем
$$
I(u^*) \le \liminf I(u_n^*) \le \lim I(u_n) = I(u).
$$
\end{proof}

\begin{thm}
Пусть $p$ четна, а $K$ выпукла.
Тогда для любой функции $u \in \Wf[-1, 1]$ выполнено $I(u^*) \le I(u)$.
\end{thm}

\begin{proof}
Обозначим $$I_{\eps}(u) = \int\limits_{-1}^1 ( 1 + | u'(x) |^2 )^{p(x) \over 2} \chi\{-1 + \eps, 1 - \eps\} dx,$$
где $\chi\{A\}$ --- характеристическая функция множества $A$, $0 < \eps < 1$.
Заметим, что для любого $0 < \eps < 1$ функция $p$, ограниченная на $[-1 + \eps, 1 - \eps]$, удовлетворяет условию (\ref{logCont}).
Тем самым, ограничив $p$ и $u$ на $[-1 + \eps, 1 - \eps]$ и применив Лемму \ref{logContIneq}, получим
$I_{\eps}(u^*) \le I_{\eps}(u)$.

Из теоремы о монотонной сходимости следует, что для любой $u \in {\Wf[-1, 1]}$ выполнено $I_{\eps}(u) \to I(u)$.
Действительно, интегранты монотонны и сходятся почти всюду к предельному.

Тем самым, получаем $I(u^*) \le I(u)$.
\end{proof}

\end{equation}
где $q(x) = p(x) - 1$,
$A(q, w) = \frac{

} \cdot {q \over q + 1}.$

\begin{lm}
Для любого $\alpha > 0$ и $q \in C^2[-1, 1]$ выпуклость функции $q(x)^\alpha$
равносильна выполнению неравенства $q''(x) \ge (1 - \alpha) {q'(x)^2 \over q(x)}$.
\end{lm}

\begin{proof}
Действительно, прямым вычислением получаем, что $(q(x)^\alpha)' = \alpha q'(x) q(x)^{\alpha - 1}$ и
$(q(x)^\alpha)'' = \alpha q''(x) q(x)^{\alpha - 1} + \alpha (\alpha - 1) q'(x)^2 q(x)^{\alpha - 2}.$
Тогда неравенство $(q(x)^\alpha)'' \ge 0$ равносильно $q''(x) \ge (1 - \alpha) {q'(x)^2 \over q(x)}$.
\end{proof}

Из леммы следует, что если $A(q, w) \le M$, где $0 < M < 1$ --- некоторая константа,
то неравенство (\ref{KconvexIneq}) следует из выпуклости функции $(p(x) - 1)^{1 - M}$.
