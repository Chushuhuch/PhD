\section{Условия, необходимые для выполнения неравенства (\ref{toprove})}

Здесь и далее мы будем ссылаться на некоторые утверждения из работы \cite{1dim},
доказательства которых в многомерном случае дословно совпадают с доказательствами в одномерном случае.

Кроме того, в данном параграфе мы для простоты опускаем параметры $x'$ и $v$ у функции $a$,
подразумевая, что все неравенства выполняются для любых $x' \in \omega, v \in \Real_+$.

\begin{prop}
{\rm \cite[Теорема 1]{1dim}.}
{\bf 1.} Если неравенство $(\ref{toprove})$ выполняется для некоторой $F \in \mathfrak{F}$
и произвольной кусочно линейной $u$, то вес $a$ четен по $y$,
то есть $a(y) \equiv a(-y)$.

{\bf 2.} Если неравенство $(\ref{toprove})$ выполняется для произвольной $F \in \mathfrak{F}$
и произвольной кусочно линейной $u$, то вес $a$ удовлетворяет неравенству
\begin{equation}
\label{almostConcave}
a( s ) + a( t ) \ge a( 1 - t + s ), \qquad -1 \le s \le t \le 1.
\end{equation}
\end{prop}

\begin{rem}
Если функция $a$ неотрицательна, а также четна и вогнута по первому аргументу, то она удовлетворяет условию $(\ref{almostConcave})$.
Действительно: для такой функции при любых $(s, t)$ выполнено $a( 1 ) - a( s ) \le a( t ) - a( -1 + t - s )$.
А так как $a( 1 ) \ge 0$, то
получаем $a( s ) + a( t ) \ge a( -1 + t - s ) = a( 1 - t + s )$.
Обратное, вообще говоря, неверно, то есть не всякая четная
неотрицательная функция, удовлетворяющая $(\ref{almostConcave})$, вогнута.
\end{rem}

\begin{rem}
\label{lanNec}
Если положить $u(\cdot, -1) \equiv 0$, то условие (\ref{almostConcave}) остается необходимым для выполнения неравенства (\ref{toprove}).
\end{rem}

\begin{prop}
\label{weightSum}
{\rm \cite[Лемма 1]{1dim}.}
Рассмотрим непрерывную функцию $a \ge 0$, заданную на $[-1,1]$
и удовлетворяющую условию (\ref{almostConcave}).
Тогда для любых $-1 \le t_1 \le t_2 \le \ldots \le t_m \le 1$ верно
\begin{align*}
\sum_{k=1}^m a(t_k) & \ge a( 1 - \sum_{k = 1}^m (-1)^k t_k ), & \text{ для четных $m$},&\\
\sum_{k=1}^m a(t_k) & \ge a( -\sum_{k = 1}^m (-1)^k t_k ),    & \text{ для нечетных $m$}.&
\end{align*}
\end{prop}
