\section{Доказательство неравенства для кусочно-ли\-ней\-ных функций в многомерном случае}

\rm
В этом параграфе мы докажем неравенство (\ref{toprove}) для кусочно линейных функций.
Не умаляя общности, будем считать, что $F(\cdot, \cdot, 0) \equiv 0$.

\begin{lm}
\label{weightSum}
{\rm \cite[Лемма 1]{1dim}.}
Рассмотрим непрерывную функцию $a \ge 0$, заданную на $[-1,1]$,
и удовлетворяющую условию (\ref{almostConcave}).
Тогда для любых $-1 \le t_1 \le t_2 \le \ldots \le t_n \le 1$ верно
\begin{align*}
\sum_{k=1}^n a(t_k) & \ge a( 1 - \sum_{k = 1}^n (-1)^k t_k ), & \text{ для четных $n$},&\\
\sum_{k=1}^n a(t_k) & \ge a( -\sum_{k = 1}^n (-1)^k t_k ),    & \text{ для нечетных $n$}.&
\end{align*}
\end{lm}

\begin{lm}
Пусть функция $a(x', \cdot, u)$ четна и удовлетворяет условию $(\ref{almostConcave})$ для всех $x'$ и $u$.
Тогда, если $u$ --- неотрицательная кусочно линейная функция, то $I(a, u) \ge I(a, u^*)$.
\end{lm}

\begin{proof}
Доказательство следует схеме доказательства теоремы 2 в \cite{1dim}.
Пусть функция $u$ имеет изломы на множестве $C \subset \Omega$.
Возьмем $$U := \{ ( x', u( x', y ) ): y \in [-1, 1] \} \setminus
\{ ( x', u( x', y ) ) : y \in [-1, 1], ( x', y ) \in C \}.$$
Тогда открытое множество $U$ разбивается в объединение конечного числа открытых множеств $G_j$.
Обозначим $m_j$ число прообразов значения $( x', u_0 ) \in G_j$, то есть число решений уравнения $u( x', y ) = u_0$
(очевидно, это число постоянно для $( x', u_0 ) \in G_j$).
Легко видеть, что эти прообразы являются линейными функциями $( x', u_0 )$:
$y = y_k^j( x', u_0 )$, $k = 1, \dots, m_j$,
и $D_n y_k^j( x', u( x', y ) ) = \frac{1}{D_n u( x', y )}$.
Мы будем считать, что $y_1^j(x', u_0) < y_2^j(x', u_0) < \dots < y_{m_j}^j(x', u_0)$.

Решение уравнения $u^*(x'_0, y^*) = u_0$ ($( x'_0, u_0 ) \in G_j$) можно выразить через $y_k^j$:

\begin{center}
\begin{tabular}{l|l|l}
\multirow{2}{*}{$u( x'_0, -1 ) < u_0$ \rule[-34pt]{0pt}{65pt}} & $m_j$ четно   & $y^* = 1-\sum\limits_{k=1}^{m_j} (-1)^k y_k^j$ \rule[-17pt]{0pt}{40pt} \\
                                                               & $m_j$ нечетно & $y^* = -\sum\limits_{k=1}^{m_j} (-1)^k y_k^j$ \rule[-17pt]{0pt}{40pt} \\ \hline
\multirow{2}{*}{$u( x'_0, -1 ) > u_0$ \rule[-34pt]{0pt}{65pt}} & $m_j$ четно   & $y^* = -1+\sum\limits_{k=1}^{m_j} (-1)^k y_k^j$ \rule[-17pt]{0pt}{40pt} \\
                                                               & $m_j$ нечетно & $y^* = \sum\limits_{k=1}^{m_j} (-1)^k y_k^j$ \rule[-17pt]{0pt}{40pt} \\
\end{tabular}
\end{center}

$\frac{1}{D_n u^*(x',y)}=\sum_{k=1}^{m_j}{|D_n y_k^j (x',u(x',y))|}$.
$D_i u^*( x', y ) = \pm \sum_{k = 1}^{m_j} ( -1 )^k D_i y_k^j( x', u( x', y ) )$, где знак перед правой частью зависит только от $j$.

Тогда 
\begin{multline*}
I(u)=\sum_{j=1}^N {
    \int_{G_j}{
        F( x', u(x), \norm{
            a_i(x',u(x)) D_i u(x), a(x,u(x)) D_n u(x)
        })
    dx}
} =
\\ \sum_{j=1}^N{
    \int_{u(G_j)}{
        \sum_{k=1}^{m_j}{
            F( x', u, \frac{
                \norm{
                    a_i(x',u) D_i y_k^j(x',u), a(x', y_k^j(x',u), u)
                }
            }{\abs{ D_n y_k^j (x',u) }})
        }
        }} {{\abs{ D_n y_k^j(x',u) }
    dx' du}
},
\end{multline*}
\begin{multline*}
I(u^*)=
\sum_{j=1}^N{
    \int_{G_j}{
        F(x', u^*, \norm{
            a_i(x',u^*(x)) D_i u^*(x), a(x,u^*(x)) D_n u^*(x)
        })
    dx}
} =
\\ \sum_{j=1}^N {
    \int\limits_{u(G_j)}{
        F( x', u^*, \frac{
            \norm{
                a_i(x',u^*) D_i y^*(x',u^*), a(x',y^*(x',u^*),u^*)
            }
        }{\sum_{k=1}^{m_j} \abs{ D_n y_k^j(x',u^*) }} )}} \cdot
        \\ \cdot {{ \sum_{k=1}^{m_j} \abs{ D_n y_k^j(x',u^*) }
    dx' du^*}
}
\end{multline*}
где $y^*=(u^*)^{-1}$.
Зафиксируем $j$, $x'$ и $u$ и докажем неравенство для подынтегральных выражений.
Обозначим
$b_k=\abs{ D_n y_k^j }$, $c_{ki}=D_i y_k^j$, $c^*_i=D_i y^*$, $y_k=y_k^j(x',u)$, $y^*=y^*(x',u)$, $m = m_j$.
Тогда требуемое утверждение имеет вид:
$$\sum_{k=1}^m{
    b_k F(\frac{ \norm{ a_ic_{ki}, a(y_k) } }{b_k})
} \ge F(
    \frac{ \norm{ a_ic^*_i, a(y^*) } }{\sum_{k=1}^m{b_k}}
)\sum_{k=1}^m{b_k}.$$

\begin{multline*}
\sum_{k=1}^m{ b_k F(\frac{ \norm{ a_ic_{ki}, a(y_k) } }{b_k}) }
\ge \text{(неравенство Йенсена) } F( \frac{ \sum_{k=1}^m \norm{ a_i c_{ki}, a(y_k)} }{ \sum_{k=1}^m b_k } ) \sum_{k=1}^m b_k \\
=   \text{(четность нормы) } F( \frac{ \sum_{k=1}^m \norm{ ( -1 )^k a_i c_{ki}, a(y_k) } }{ \sum_{k=1}^m b_k} ) \sum_{k=1}^m b_k \\
\ge \text{(неравенство треугольника) } F( \frac{ \norm{ \sum_{k = 1}^m ( -1 )^k a_i c_{ki}, \sum_{k = 1}^m a( y_k ) } }{ \sum_{k=1}^m b_k} ) \sum_{k=1}^m b_k \\
\ge \text{(лемма (\ref{weightSum})) } F( \frac{ \norm{ \sum_{k = 1}^m ( -1 )^k a_i c_{ki}, a( y^* ) } }{ \sum_{k=1}^m b_k} ) \sum_{k=1}^m b_k \\
=   \text{(четность нормы) } F( \frac{ \norm{ \pm a_i \sum_{k = 1}^m ( -1 )^k c_{ki}, a( y^* ) } }{ \sum_{k=1}^m b_k} ) \sum_{k=1}^m b_k
= F( \frac{ \norm{ a_i c^*_i, a(y^*) } }{\sum_{k=1}^m b_k} ) \sum_{k=1}^m b_k.
\end{multline*}


С помощью неравенства Йенсена для функции $F$
получаем $$T \ge F(
    \frac{\sum_{k=1}^m \norm{ a_i c_{ki}, a(y_k)}} {\sum_{k=1}^m{b_k}}
)\sum_{k=1}^m{b_k}.$$ Тогда нам надо доказать $$\sum_{k=1}^m
\norm{ a_i c_{ki}, a(y_k) } \ge \norm{ a_i c^*_{i}, a(y^*)
}=\bignorm{ \pm a_i \sum_{k=1}^m{(-1)^k c_{ki}}, a(y^*) }.$$

Заметим, что под знаком нормы можно поменять все координаты
кроме последней на противоположные (поменяв сначала все, а потом
последнюю). Получаем
$$P:=\sum_{k=1}^m
\norm{ ( -1 )^k a_i c_{ki}, a(y_k) } \ge \bignorm{ a_i \sum_{k=1}^m{(-1)^k c_{ki}}, a(y^*) }.$$
По неравенству треугольника
$P \ge \bignorm{ \sum_{k = 1}^m ( -1 )^k a_i c_{ki}, \sum_{k = 1}^m a( y_k ) }.$
А по лемме для последней координаты $\sum_{k = 1}^m a( y_k ) \ge a( y^* )$.
Норма, симметричная по последней координате, монотонна по ней. Тем самым, получили требуемое.
\end{proof}
