\section{Введение}
Пусть $\Omega = \omega \times [-1,1] \subset \Real^n$.
Обозначим $x = ( x_1, \dots, x_{n - 1}, y ) = ( x', y )$.
Пусть $\norm{\cdot}$ --- некоторая норма в $\Real^n$, симметричная по последней координате.
$i$ всегда меняется от $1$ до $n - 1$.
$D_k$ будем обозначать частную производную по $k$-ой координате.
Рассмотрим измеримую неотрицательную функцию $u$, заданную на $\Omega$.
Для нее верна формула о послойном представлении.
Именно, положим $\mathcal{A}_t(x') := \{ y \in [-1,1] :\ u( x', y ) > t \}$.
Тогда имеет место равенство $u(x', y) = \int_0^\infty \chi_{\mathcal{A}_t(x')} dt$.

Определим монотонную перестановку измеримого множества $E \subset [-1, 1]$ и
монотонную перестановку неотрицательной функции $u \in \W(\Omega)$:
\begin{eqnarray*}
E^* := [1 - \abs{E}, 1] \qquad
u^*(x', y) = \int_0^\infty \chi_{\mathcal{A}_t(x')^*} dt
\end{eqnarray*}

Определим множество $\mathfrak{F}$ непрерывных функций $F: \omega \times \Real_+ \times \Real_+ \to \Real_+$,
выпуклых и возрастающих по третьему аргументу.

Рассмотрим функционал:
\begin{equation}
\label{functional}
I( a, u ) = \int_\Omega F( x', u(x), \norm{ a_i( x', u( x ) ) D_i u( x ), a( x, u( x ) ) D_n u( x ) } ) dx,
\end{equation}
где $a( \cdot, \cdot ): \Omega \times \Real_+ \to \Real_+$ и $a_i( \cdot, \cdot ): \omega \times \Real_+ \to \Real$ --- непрерывные функции,
$F \in \mathfrak{F}$.

Хорошо известно, что при $a \equiv const$ справедливо неравенство
\begin{eqnarray}
\label{toprove}
I(a, u^*) &\le& I(a, u), \qquad \qquad u \in \W(-1, 1); \\
\end{eqnarray}
(см., например, \cite{Kawohl} и цитированную там литературу).
Кроме того, одномерный случай данного неравенства доказан нами в работе \cite{1dim}.

\section{Условия, необходимые для выполнения неравенства (\ref{toprove})}

Необходимые условия обсуждались нами в работе \cite{1dim}. Приведем здесь лишь формулировки.

\begin{thm}
{\bf 1} Если неравенство $(\ref{toprove})$ выполняется для некоторой $F \in \mathfrak{F}$
и произвольной кусочно линейной $u$, то вес $a$ четен по $y$,
то есть $a(x', y, v) \equiv a(x', -y, v)$.

{\bf 2} Если неравенство $(\ref{toprove})$ выполняется для произвольной $F \in \mathfrak{F}$
и произвольной кусочно линейной $u$, то вес $a$ удовлетворяет неравенству
\begin{equation}
\label{almostConcave}
a( x', s, v ) + a( x', t, v ) \ge a( x', 1 - t + s, v ), \qquad x \in \omega, -1 \le s \le t \le 1, v \in \Real_+.
\end{equation}
\end{thm}

\begin{rem}
Если функция $a$ неотрицательна, а также четна и вогнута по первому параметру, то она удовлетворяет условию $(\ref{almostConcave})$.
Действительно: для любых $s$, $t$ и $u$ выполнено $a( 1, u ) - a( s, u ) \le a( t, u ) - a( -1 + t - s, u )$.
А так как $a( 1, u ) \ge 0$, то
получаем $a( s, u ) + a( t, u ) \ge a( -1 + t - s, u ) = a( 1 - t + s, u )$.
Обратное вообще говоря неверно, то есть не всякая четная
неотрицательная функция, удовлетворяющая $(\ref{almostConcave})$, вогнута.
\end{rem}

\section{Доказательство неравенства для кусочно-ли\-ней\-ных функций в многомерном случае}

\begin{prop}
Теорема $(\ref{linth})$ верна в многомерном случае в тех же предположениях относительно функции $a$ и
для всех кусочно-линейных $u$ таких, что при фиксированном $x'$ $u( x', \cdot )$ принимает конечное число значений.
\end{prop}

\begin{proof}
Аналогично введем множества $G_j$,
функции $y_k^j(x',u)$, $u \in u(G_j)$: $(x',y_k^j(x',u))\in G_j$ и
$u(x',y_k^j(x',u))=u$. И
$D_n y_k^j (x',u(x',y))=\frac{1}{D_n u(x',y)}$.

Опять, если $u^*(x'_0,y^*)=u_0$ и $u(x'_0,y_k^j)=u_0$, где $u_0\in u(G_j)$,
то при $u(x'_0,-1)<u_0$ и четном $m_j$
$y^*=1-\sum_{k=1}^{m_j}{(-1)^k y_k^j}$, тогда как при нечетном
$y^*=-\sum_{k=1}^{m_j}{(-1)^k y_k^j}$. Если $u(x'_0,-1)>u_0$,
получится с обратным знаком.
$\frac{1}{D_n u^*(x',y)}=\sum_{k=1}^{m_j}{|D_n y_k^j (x',u(x',y))|}$.
$D_i u^*( x', y ) = \pm \sum_{k = 1}^{m_j} ( -1 )^k D_i y_k^j$.

Тогда 
\begin{multline*}
I(u)=\sum_{j=1}^N {
    \int_{G_j}{
        F( x', u(x), \norm{
            a_i(x',u(x)) D_i u(x), a(x,u(x)) D_n u(x)
        })
    dx}
} =
\\ \sum_{j=1}^N{
    \int_{u(G_j)}{
        \sum_{k=1}^{m_j}{
            F( x', u, \frac{
                \norm{
                    a_i(x',u) D_i y_k^j(x',u), a(x', y_k^j(x',u), u)
                }
            }{\abs{ D_n y_k^j (x',u) }})
        }
        }} {{\abs{ D_n y_k^j(x',u) }
    d(x',u)}
},
\end{multline*}
\begin{multline*}
I(u^*)=
\sum_{j=1}^N{
    \int_{G_j}{
        F(x', u^*, \norm{
            a_i(x',u^*(x)) D_i u^*(x), a(x,u^*(x)) D_n u^*(x)
        })
    dx}
} =
\\ \sum_{j=1}^N {
    \int\limits_{u(G_j)}{
        F( x', u^*, \frac{
            \norm{
                a_i(x',u^*) D_i y^*(x',u^*), a(x',y^*(x',u^*),u^*)
            }
        }{\sum_{k=1}^{m_j} \abs{ D_n y_k^j(x',u^*) }} )}} \cdot
        \\ \cdot {{ \sum_{k=1}^{m_j} \abs{ D_n y_k^j(x',u^*) }
    d(x',u^*)}
}
\end{multline*}
 где $y^*=(u^*)^{-1}$. Зафиксируем $j$, $x'$ и $u$ и
докажем неравенство для подынтегральных выражений. Обозначим
$b_k=\abs{ D_n y_k^j }$, $c_{ki}=D_i y_k^j$, $c^*_i=D_i y^*$,
$y_k=y_k^j(x',u)$, $y^*=y^*(x',u)$, $m = m_j$. Тогда требуемое
утверждение имеет вид: $$T=\sum_{k=1}^m{
    b_k F(\frac{ \norm{ a_ic_{ki}, a(y_k) } }{b_k})
} \ge F(
    \frac{ \norm{ a_ic^*_i, a(y^*) } }{\sum_{k=1}^m{b_k}}
)\sum_{k=1}^m{b_k}$$ С помощью неравенства Йенсена для функции $F$
получаем $$T \ge F(
    \frac{\sum_{k=1}^m \norm{ a_i c_{ki}, a(y_k)}} {\sum_{k=1}^m{b_k}}
)\sum_{k=1}^m{b_k}.$$ Тогда нам надо доказать $$\sum_{k=1}^m
\norm{ a_i c_{ki}, a(y_k) } \ge \norm{ a_i c^*_{i}, a(y^*)
}=\bignorm{ \pm a_i \sum_{k=1}^m{(-1)^k c_{ki}}, a(y^*) }.$$

Заметим, что под знаком нормы можно поменять все координаты
кроме последней на противоположные (поменяв сначала все, а потом
последнюю). Получаем
$$P:=\sum_{k=1}^m
\norm{ ( -1 )^k a_i c_{ki}, a(y_k) } \ge \bignorm{ a_i \sum_{k=1}^m{(-1)^k c_{ki}}, a(y^*) }.$$
По неравенству треугольника
$P \ge \bignorm{ \sum_{k = 1}^m ( -1 )^k a_i c_{ki}, \sum_{k = 1}^m a( y_k ) }.$
А по лемме для последней координаты $\sum_{k = 1}^m a( y_k ) \ge a( y^* )$.
Норма, симметричная по последней координате, монотонна по ней. Тем самым, получили требуемое.
\end{proof}
