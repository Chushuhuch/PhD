\section{Введение}
Пусть $\Omega = \omega \times [-1,1]$,
где $\omega$ --- ограниченная область в $\Real^{n - 1}$.
Обозначим $x = ( x_1, \dots, x_{n - 1}, y ) = ( x', y )$.
Напомним теорему о послойном представлении измеримой неотрицательной функции $u$, заданной на $\Omega$
(см. \cite[Теорему 1.13]{LL}).
Положим $\mathcal{A}_t(x') := \{ y \in [-1,1] :\ u( x', y ) > t \}$.
Тогда имеет место равенство $u(x', y) = \int_0^\infty \mathcal{X}\{\mathcal{A}_t(x')\}(y) dt$,
где $\mathcal{X}\{A\}$ --- характеристическая функция множеста $A$.

Определим монотонную перестановку измеримого множества $E \subset [-1, 1]$ и
монотонную перестановку неотрицательной функции $u \in \W(\Omega)$:
\begin{eqnarray*}
E^* := [1 - \abs{E}, 1]; \qquad
u^*(x', y) = \int_0^\infty \mathcal{X}\{\mathcal{A}^*_t(x')\}(y) dt.
\end{eqnarray*}

Определим множество $\mathfrak{F}$ непрерывных функций $F: \omega \times \Real_+ \times \Real_+ \to \Real_+$
(здесь и далее $\Real_+ = [0, \infty)$), выпуклых и возрастающих по третьему аргументу.

Рассмотрим функционал:
\begin{equation}
\label{functional}
I( a, u ) = \int_\Omega F( x', u(x), \norm{ D_a u } ) dx,
\end{equation}
где $F \in \mathfrak{F}$,
$\norm{\cdot}$ --- некоторая норма в $\Real^n$, симметричная по последней координате,
$$D_a u = ( a_1( x', u( x ) ) D_1 u, \dots, a_{n - 1}( x', u( x ) ) D_{n - 1} u, a( x, u( x ) ) D_n u )$$
--- градиент $u$ с весом (обратите внимание, что только вес при $D_n u$ зависит от $y$),
$a( \cdot, \cdot ): \Omega \times \Real_+ \to \Real_+$ и $a_i( \cdot, \cdot ): \omega \times \Real_+ \to \Real_+$ --- непрерывные ограниченные функции.
Здесь и далее индекс $i$ пробегает от $1$ до $n - 1$.

Хорошо известно, что при $a \equiv const$ справедливо неравенство
\begin{equation}
\label{toprove}
I(a, u^*) \le I(a, u), \qquad \qquad u \in \W(\Omega), u \ge 0
\end{equation}
(см., например, \cite{Kawohl} и цитированную там литературу).

В настоящей работе мы приводим необходимые условия на вес $a$ для выполнения неравенства (\ref{toprove})
и доказываем неравенство (\ref{toprove}) при некоторых дополнительных ограничениях.

В работе \cite{Br} рассматривается неравенство, аналогичное (\ref{toprove}).
А именно,
\begin{equation}
\label{breq}
I(a, \overline{u}) \le I(a, u), \qquad \qquad u \in \W(\Real^n), u \ge 0,
\end{equation}
где $\overline{u}$ --- симметризация по Штейнеру (симметричная перестановка) функции $u$ по переменной $y$.
Автор \cite{Br} доказывает неравенство в случае, когда вес $a$ --- четная и выпуклая по $y$ функция.
Для этого неравенство (\ref{breq}) доказывается для кусочно линейных функций $u$
и утверждается, что в общем случае неравенство получается при помощи аппроксимации.
Однако предельный переход в \cite{Br} должным образом не обоснован,
и можно считать, что результат получен лишь для липшицевых функций $u$.

В одномерном случае неравенства (\ref{toprove}) и (\ref{breq}) рассматриваются в работе \cite{1dim},
где получены необходимые и достаточные условия их выполнения.
В частности, пробел в работе \cite{Br} был закрыт для $n = 1$.

В статье \cite{Lan} рассматривается неравенство (\ref{toprove}) с аналогичным функционалом в двумерном случае
при дополнительном ограничении $u(\cdot, -1) \equiv 0$.
Отметим, что наши условия на весовые функции слабее, чем в \cite{Lan}.

Статья разделена на четыре параграфа.
В \S2 приведены условия, необходимые для выполнения неравенства (\ref{toprove}).
В \S3 приведено доказательство неравенства (\ref{toprove}) для кусочно линейных функций $u$.
\S4 содержит доказательство неравенства в более общих случаях.

\section{Условия, необходимые для выполнения неравенства (\ref{toprove})}

Здесь и далее мы будем ссылаться на некоторые утверждения из работы \cite{1dim},
доказательства которых в многомерном случае дословно совпадают с доказательствами в одномерном случае.

Кроме того, в данном параграфе мы для простоты опускаем параметры $x'$ и $v$ у функции $a$,
подразумевая, что все неравенства выполняются для любых $x' \in \omega, v \in \Real_+$.

\begin{prop}
{\rm \cite[Теорема 1]{1dim}.}
{\bf 1.} Если неравенство $(\ref{toprove})$ выполняется для некоторой $F \in \mathfrak{F}$
и произвольной кусочно линейной $u$, то вес $a$ четен по $y$,
то есть $a(y) \equiv a(-y)$.

{\bf 2.} Если неравенство $(\ref{toprove})$ выполняется для произвольной $F \in \mathfrak{F}$
и произвольной кусочно линейной $u$, то вес $a$ удовлетворяет неравенству
\begin{equation}
\label{almostConcave}
a( s ) + a( t ) \ge a( 1 - t + s ), \qquad -1 \le s \le t \le 1.
\end{equation}
\end{prop}

\begin{rem}
Если функция $a$ неотрицательна, а также четна и вогнута по первому аргументу, то она удовлетворяет условию $(\ref{almostConcave})$.
Действительно: для такой функции при любых $(s, t)$ выполнено $a( 1 ) - a( s ) \le a( t ) - a( -1 + t - s )$.
А так как $a( 1 ) \ge 0$, то
получаем $a( s ) + a( t ) \ge a( -1 + t - s ) = a( 1 - t + s )$.
Обратное, вообще говоря, неверно, то есть не всякая четная
неотрицательная функция, удовлетворяющая $(\ref{almostConcave})$, вогнута.
\end{rem}

\begin{rem}
\label{lanNec}
Если положить $u(\cdot, -1) \equiv 0$, то условие (\ref{almostConcave}) остается необходимым для выполнения неравенства (\ref{toprove}).
\end{rem}

\begin{prop}
\label{weightSum}
{\rm \cite[Лемма 1]{1dim}.}
Рассмотрим непрерывную функцию $a \ge 0$, заданную на $[-1,1]$
и удовлетворяющую условию (\ref{almostConcave}).
Тогда для любых $-1 \le t_1 \le t_2 \le \ldots \le t_m \le 1$ верно
\begin{align*}
\sum_{k=1}^m a(t_k) & \ge a( 1 - \sum_{k = 1}^m (-1)^k t_k ), & \text{ для четных $m$},&\\
\sum_{k=1}^m a(t_k) & \ge a( -\sum_{k = 1}^m (-1)^k t_k ),    & \text{ для нечетных $m$}.&
\end{align*}
\end{prop}

\section{Доказательство неравенства (\ref{toprove}) для кусочно ли\-ней\-ных функций}

В этом параграфе мы докажем неравенство (\ref{toprove}) для кусочно линейных функций.
Не умаляя общности, будем считать, что $F(\cdot, \cdot, 0) \equiv 0$.

\begin{lm}
Пусть функция $a(x', \cdot, u)$ четна и удовлетворяет условию $(\ref{almostConcave})$ для всех $(x', u)$.
Тогда, если $u$ --- неотрицательная кусочно линейная функция, то $I(a, u) \ge I(a, u^*)$.
\end{lm}

\begin{proof}
Доказательство следует схеме доказательства теоремы 2 в \cite{1dim}.
Пусть функция $u$ имеет изломы на множестве $C \subset \Omega$.
Возьмем $$U := \{ ( x', u( x', y ) ): y \in (-1, 1), (x', y) \not\in C \}.$$
Тогда открытое множество $U$ разбивается в объединение конечного числа связных открытых множеств $G_j$.
Обозначим $m_j$ число прообразов значения $( x', u_0 ) \in G_j$, то есть число решений уравнения $u( x', y ) = u_0$
(очевидно, это число постоянно для $( x', u_0 ) \in G_j$).
Легко видеть, что эти прообразы являются линейными функциями $( x', u_0 )$:
$y = y_k^j( x', u_0 )$, $k = 1, \dots, m_j$,
и $D_n y_k^j( x', u( x', y ) ) = \frac{1}{D_n u( x', y )}$.
Мы будем считать, что $y_1^j(x', u_0) < y_2^j(x', u_0) < \dots < y_{m_j}^j(x', u_0)$.

Уравнение $u^*(x'_0, y^*) = u_0$ задает $y^*$ как функцию $( x'_0, u_0 ) \in G_j$.
Ее можно выразить через $y_k^j$ (в частности, $y^*$ кусочно линейна):

\begin{center}
\begin{tabular}{l|l|l}
\multirow{2}{*}{$u( x'_0, -1 ) < u_0$ \rule[-34pt]{0pt}{65pt}} & $m_j$ четно   & $y^* = 1-\sum\limits_{k=1}^{m_j} (-1)^k y_k^j$ \rule[-17pt]{0pt}{40pt} \\
                                                               & $m_j$ нечетно & $y^* = -\sum\limits_{k=1}^{m_j} (-1)^k y_k^j$ \rule[-17pt]{0pt}{40pt} \\ \hline
\multirow{2}{*}{$u( x'_0, -1 ) > u_0$ \rule[-34pt]{0pt}{65pt}} & $m_j$ четно   & $y^* = -1+\sum\limits_{k=1}^{m_j} (-1)^k y_k^j$ \rule[-17pt]{0pt}{40pt} \\
                                                               & $m_j$ нечетно & $y^* = \sum\limits_{k=1}^{m_j} (-1)^k y_k^j$ \rule[-17pt]{0pt}{40pt} \\
\end{tabular}
\end{center}

Отсюда ясно, что
\begin{equation*}
D_n y^*(x', u(x', y)) = \frac{1}{D_n u^*(x',y)}=\sum_{k=1}^{m_j}{|D_n y_k^j (x',u(x',y))|}
\end{equation*}
и $D_i u^*( x', y ) = \pm \sum_{k = 1}^{m_j} ( -1 )^k D_i y_k^j( x', u( x', y ) )$, где знак перед правой частью зависит только от $j$.

Тогда 
\begin{multline}
\label{intu}
I(u)=\sum_{j=1}^N {
    \int\limits_{G_j}{
        F( x', u(x), \norm{
            a_i(x',u(x)) D_i u(x), a(x,u(x)) D_n u(x)
        })
    dx}
}
\\ = \sum_{j=1}^N{
    \int\limits_{u(G_j)}{
        \sum_{k=1}^{m_j}{
            F \Big( x', u, \frac{
                \norm{
                    a_i(x',u) D_i y_k^j(x',u), a(x', y_k^j(x',u), u)
                }
            }{\abs{ D_n y_k^j (x',u) }} \Big)
        }
        }} {{\abs{ D_n y_k^j(x',u) }
    dx' du}
},
\end{multline}
\begin{multline}
\label{intus}
I(u^*)=
\sum_{j=1}^N{
    \int\limits_{G_j}{
        F(x', u^*, \norm{
            a_i(x',u^*(x)) D_i u^*(x), a(x,u^*(x)) D_n u^*(x)
        })
    dx}
}
\\ = \sum_{j=1}^N {
    \int\limits_{u(G_j)}{
        F \Big( x', u^*, \frac{
            \norm{
                a_i(x',u^*) D_i y^*(x',u^*), a(x',y^*(x',u^*),u^*)
            }
        }{\sum_{k=1}^{m_j} \abs{ D_n y_k^j(x',u^*) }} \Big) }} \times
        \\ \times {{ \sum_{k=1}^{m_j} \abs{ D_n y_k^j(x',u^*) }
    dx' du^*}
}.
\end{multline}
Зафиксируем $j$, $x'$ и $u$ и обозначим
$b_k=\abs{ D_n y_k^j }$, $c_{ki}=D_i y_k^j$, $c^*_i=D_i y^*$, $y_k=y_k^j(x',u)$, $y^*=y^*(x',u)$, $m = m_j$.
Тогда справедлива следующая цепочка неравенств:

\begin{multline}
\label{chain}
\sum_{k=1}^m{ b_k F\Big(\frac{ \norm{ a_ic_{ki}, a(y_k) } }{b_k}\Big) }
\overset{a}{\ge} F\Big( \frac{ \sum_{k=1}^m \norm{ a_i c_{ki}, a(y_k)} }{ \sum_{k=1}^m b_k } \Big) \sum_{k=1}^m b_k \\
\overset{b}{=}  F\Big( \frac{ \sum_{k=1}^m \norm{ ( -1 )^k a_i c_{ki}, a(y_k) } }{ \sum_{k=1}^m b_k} \Big) \sum_{k=1}^m b_k
\overset{c}{\ge}  F\Big( \frac{ \norm{ \sum_{k = 1}^m ( -1 )^k a_i c_{ki}, \sum_{k = 1}^m a( y_k ) } }{ \sum_{k=1}^m b_k} \Big) \sum_{k=1}^m b_k \\
\overset{d}{\ge} F\Big( \frac{ \norm{ \sum_{k = 1}^m ( -1 )^k a_i c_{ki}, a( y^* ) } }{ \sum_{k=1}^m b_k} \Big) \sum_{k=1}^m b_k \\
\overset{e}{=}   F\Big( \frac{ \norm{ \pm a_i \sum_{k = 1}^m ( -1 )^k c_{ki}, a( y^* ) } }{ \sum_{k=1}^m b_k} \Big) \sum_{k=1}^m b_k
= F\Big( \frac{ \norm{ a_i c^*_i, a(y^*) } }{\sum_{k=1}^m b_k} \Big) \sum_{k=1}^m b_k.
\end{multline}
Здесь в переходе (a) применено неравенство Йенсена, в переходах (b) и (e) использована четность нормы, в (c) использовано неравенство треугольника,
в (d) --- предложение \ref{weightSum} и четность веса $a$ по $y$.

Из (\ref{chain}) видно, что подынтегральное выражение в (\ref{intu}) не меньше
подынтегрального выражения в (\ref{intus}).
Тем самым, доказательство завершено.
\end{proof}

\begin{rem}
\label{lanLin}
Если $u(\cdot, -1) \equiv 0$, то утверждение леммы верно без условия четности веса.
Четность используется только в переходе (d) цепочки неравенств (\ref{chain}),
и, поскольку при $u(\cdot, -1) \equiv 0$ всегда выполнено $u(x'_0, -1) < u_0$,
то предложение \ref{weightSum} как раз обеспечивает требуемые для перехода (d) неравенства.
\end{rem}

\section{Переход к соболевским функциям}

Следующие утверждения доказаны в одномерном случае в работе \cite{1dim}.
Доказательства в многомерном случае совершенно аналогичны.
\begin{prop} {\rm \cite[Лемма 5]{1dim}.}
Функционал $I(a, u)$ слабо полунепрерывен снизу в $W^1_1(\Omega)$.
\end{prop}

\begin{prop}
\label{uplift}
{\rm \cite[Лемма 6]{1dim}.}
Пусть $A \subset W^1_1(\Omega)$. И пусть $B \subset A$ таково, что
$\forall v \in B$ выполнено $I(a, v^*) \le I(a, v)$. Предположим, что для каждого $u \in A$
найдется последовательность $u_k \in B$ такая, что $u_k \to u$ в $W^1_1(\Omega)$ и
$I(a, u_k) \to I(a, u)$. Тогда $\forall u \in A$ будет выполнено $I(a, u^*) \le I(a, u)$.
\end{prop}

\begin{thm}
\label{mainThm}
Пусть функция $a(x', \cdot, u)$ четна и удовлетворяет условию $(\ref{almostConcave})$ для всех $x'$ и $u$.

{\bf 1.} Неравенство (\ref{toprove}) верно для произвольной неотрицательной $u \in Lip(\Omega)$.

{\bf 2.} Предположим, что $\partial \omega \in Lip$ и
для любых $x' \in \omega, z \in \Real_+, p \in \Real$
функция $F$ удовлетворяет неравенству
$$F( x', z, p ) \le C ( 1 + |z|^{q^*} + |p|^q ),$$
где $\frac{1}{q^*} = \frac{1}{q} - \frac{1}{n}$, если $q < n$, либо $q^*$ любое в противном случае.
Если $q \le n$, то дополнительно предположим, что веса $a$ и $a_i$ ограничены.
Тогда неравенство (\ref{toprove}) верно для произвольной неотрицательной $u \in W^1_q(\Omega)$.
\end{thm}
\begin{proof}
{\bf 1.} Мы можем приблизить липшицевы $u$ кусочно линейными функциями $u_k$ вместе с производными почти всюду.
Поскольку $u_k$ равномерно ограничены вместе с производными,
то и $F(x', u_k(x), \norm{D_a u_k})$ равномерно ограничены.
Тогда мы можем воспользоваться теоремой Лебега, получив $u_k \to u$ в $W^1_1(\Omega)$ и $I(a, u_k) \to I(a, u)$.
Воспользовавшись предложением \ref{uplift}, получаем требуемое.

{\bf 2.} Рассмотрим произвольную $u \in W^1_q(\Omega)$.
Для нее можно построить последовательность кусочно линейных функций $u_k$, приближающих ее в $W^1_q(\Omega)$.
Например, этого можно добиться, приблизив $u$ гладкими функциями, которые, в свою очередь приблизить кусочно линейными.

Тогда, ввиду предложения \ref{uplift}, достаточно добиться $I(a, u_k) \to I(a, u)$.

Переходя к подпоследовательности, можем считать, что $u_k \to u$ и $\nabla u_k \to \nabla u$ почти всюду.
Покажем, что веса $a_i(x', u(x))$ и $a(x, u(x))$ ограничены.
Если $q \le n$, то это выполнено по предположению теоремы. Если же нет, то $W^1_q(\Omega)$ вкладывается в $C(\Omega)$,
тем самым, $u_k(x)$ равномерно ограничены, а значит, и $a_i(x', u_k(x))$ и $a(x, u_k(x))$ равномерно ограничены.
Поэтому $\norm{ D_a u_k( x ) } \le C_1 |\nabla u_k( x )|$.
То есть,
$$F(x', u_k(x), \norm{ D_a u_k( x ) }) \le C \cdot C_1^q ( 1 + |u_k(x)|^{q^*} + |\nabla u_k|^q ).$$ % \le 2 C \cdot C_1^q ( 1 + |u(x)|^{q^*} + |\nabla u( x )|^q ).$$

Рассмотрим множества $A_K$, состоящие из $x$, для которых при всех $k \ge K$ выполнено
$C_1^q ( 1 + |u_k(x)|^{q^*} + |\nabla u_k( x )|^q ) \le 2 C \cdot C_1^q ( 1 + |u(x)|^{q^*} + |\nabla u( x )|^q ) \}$.
Очевидно, что $A_K \subset A_{K + 1}$. Кроме того, поскольку $u_k$ сходятся почти всюду вместе с производными к $u$, то
$|A_K| \to |\Omega|$. Введем множества $B_K = A_K \setminus A_{K - 1}$ ($B_1 = A_1$).
По теореме вложения $\Vert u_k \Vert_{q^*} \le C_2 \Vert u_k \Vert_{W^1_q}$.
Тогда на любом $A_K$ и, следовательно, $B_K$, у $F(x', u_k(x), \norm{ D_a u_k })$ есть суммируемая мажоранта и
$$\int_{B_K} F(x', u_k(x), \norm{ D_a u_k }) dx \to \int_{B_K} F(x', u(x), \norm{ D_a u }) dx.$$
Имеем
\begin{multline*}
I( a, u ) = \sum_K \int_{B_K} F(x', u(x), \norm{ D_a u }) dx =
 \sum_K \lim\limits_{k \to \infty} \int_{B_K} F(x', u_k(x), \norm{ D_a u_k }) dx \\ =
 \lim\limits_{k \to \infty} \sum_K \int_{B_K} F(x', u_k(x), \norm{ D_a u_k }) dx =
 \lim\limits_{k \to \infty} I( a, u_k ).
\end{multline*}
Тем самым, сходимость $I(a, u_k) \to I(a, u)$ доказана.
\end{proof}

Аналогично, с учетом замечаний (\ref{lanNec}) и (\ref{lanLin}), доказывается
\begin{thm}
Пусть $u(\cdot, -1) \equiv 0$ и функция $a(x', \cdot, u)$ удовлетворяет условию $(\ref{almostConcave})$ для всех $x'$ и $u$.
Тогда верны выводы теоремы \ref{mainThm}.
\end{thm}
