\section{Условия, необходимые для выполнения неравенства (\ref{toprove})}

\begin{thm}
{\bf 1} Если неравенство (\ref{toprove}) выполняется для некоторой $F \in \mathfrak{F}$
и произвольной кусочно линейной $u$, то вес $a$ четен по первому аргументу,
то есть $a(x, v) \equiv a(-x, v)$.
{\bf 2} Если неравенство (\ref{toprove}) выполняется для произвольной $F \in \mathfrak{F}$
и произвольной кусочно линейной $u$, то вес $a$ удовлетворяет неравенству
\begin{equation}
\label{almostConcave}
\forall -1 \le s \le t \le 1, \forall v \quad a( s, v ) + a( t, v ) \ge a( 1 - t + s, v ).
\end{equation}
\end{thm}

\begin{proof}
{\bf 1.} Пусть $a(\bar{x}, \bar{v}) < a(-\bar{x}, \bar{v})$, $\abs{\bar{x}} \neq 1$.
Тогда
$$\exists \eps : \abs{ x - \bar{x} } < \eps, \abs{ v - \bar{v} } < \eps \Rightarrow a(x, v) < a(-x, v),$$
и можно взять следующую функцию: $u([-1,\bar{x}-\eps]) = \bar{v} + \eps$,
$u([\bar{x}+\eps, 1]) = \bar{v} - \eps$, $u(x) = \bar{v} + \bar{x} - x$ при
$\abs{ x-\bar{x} } < \eps$. Тогда $u^*(x, v) = u(-x, v)$ и
\begin{multline*}
I(u)-I(u^*) = \int_{\bar{x}-\eps}^{\bar{x}+\eps} F( \bar{v} + \bar{x} - x, a(x, \bar{v} + \bar{x} - x) ) dx -
\int_{-\bar{x}-\eps}^{-\bar{x}+\eps} F( \bar{v} + \bar{x} + x, a(x, \bar{v} + \bar{x} + x) ) dx \\ =
\int_{\bar{x}-\eps}^{\bar{x}+\eps} ( F( \bar{v} + \bar{x} - x, a(x, \bar{v} + \bar{x} - x) ) -
F( \bar{v} + \bar{x} - x, a(-x, \bar{v} + \bar{x} - x) ) ) dx < 0,
\end{multline*}
что противоречит условию.

{\bf 2.} Предположим, что условие (\ref{toprove}) не выполняется.
Тогда найдутся $-1 \le s \le t \le 1$, $\bar{v}$ и $\delta$, такие, что
$a( s, \bar{v} ) + a( t, \bar{v} ) + 2 \delta < a( 1 - t + s, \bar{v} )$.
Тогда найдется $\eps > 0$ такое, что для произвольного $0 \le y \le \eps$ и $\bar{v} \le v \le \bar{v} + \eps$ останется
$a( s + y, v ) + a( t - y, v ) + \delta < a( 1 - t + s + 2y, v )$.

Рассмотрим функцию $u$:
$u( [-1, s] \cup [t, 1] ) = \bar{v}$, $u( [s + \eps, t - \eps] ) = \bar{v} + \eps$,
$u( x ) = \bar{v} + x - s$ при $x \in [s, s + \eps]$,
$u( x ) = \bar{v} + t - x$ при $x \in [t - \eps, t]$.

$u^*( x ) = \bar{v} + \frac{ x - ( 1 - t + s ) }{2}$ при $x \in [1 - t + s, 1 - t + s + 2\eps]$.

Тогда получим
\begin{multline*}
I(u^*) = \int_0^{2\eps} F(u(1 - t + s + z), \frac{a(1 - t + s + z, u(1 - t + s + z))}{2}) dz =\\
\int_0^\eps 2 F(\bar{v} + y, \frac{a(1 - t + s + 2y, \bar{v} + y)}{2}) dy\\
0 \le I( u ) - I( u^* ) =
\int_0^\eps ( F(\bar{v} + y, a(s + y, \bar{v} + y)) + F(\bar{v} + y, a( t - y, \bar{v} + y)) -\\
2 F(\bar{v} + y, \frac{ a(1 - t + s + 2y, \bar{v} + y) }{2})) dy <
\\< \int_0^\eps ( F(\bar{v} + y, a(s + y, \bar{v} + y)) + F(\bar{v} + y, a(t - y, \bar{v} + y)) -\\
2 F(\bar{v} + y, \frac{ a(s + y, \bar{v} + y) + a(t - y, \bar{v} + y) + \delta }{2})) dy =: J
\end{multline*}

Положим теперь $F(v, p) = p^\alpha$. Такая функция строго выпукла при $\alpha > 1$.
Однако по заданному $\delta$ можно найти такое $\alpha$,
чтобы $\frac{F(v, p) + F(v, q)}{2} \le F(v, \frac{p + q}{2} + \frac{\delta}{2})$ при $p, q \in [0, A]$.
Для этого решим неравенство
$\frac{ p^\alpha + q^\alpha }{2} \le ( \frac{p + q}{2} + \frac{\delta}{2} )^\alpha$
относительно $\frac{\delta}{2}$. Получим
$\frac{\delta}{2} \ge (\frac{p^\alpha + q^\alpha}{2} )^{\frac{1}{\alpha}} - \frac{p + q}{2}$.
Максимум последнего выражения достигается, например, при $p = 0$, $q = A$, то есть
$\frac{\delta}{2} \ge A(\frac{1}{2^{\frac{1}{\alpha}}} - \frac{1}{2})$.
Итого, можно взять $\alpha = -( \log_2 ( \frac{\delta}{2 A} + \frac{1}{2} ) )^{-1} > 1$.

Тогда, если взять $A = \max a$, $\alpha = -( \log_2 ( \frac{\delta}{2 A} + \frac{1}{2} ) )^{-1}$
и $F(v, p) = p^\alpha$, получим $J \le 0$ и, тем самым, противоречие.
\end{proof}

\rem Видно, что в доказательстве второго пункта теоремы
функцию $u$ на отрезке $[-1, s]$ можно заменить на любую возрастающую функцию.
Тем самым, условие (\ref{almostConcave}) является необходимым и для выполнения неравенства (\ref{toprove})
в случае закрепленных на левом конце функций: $u( -1 ) = 0$.
