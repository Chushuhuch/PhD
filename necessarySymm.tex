\section{Дополнение. Необходимые условия в случае симметричной перестановки}

Будем рассматривать условия, необходимые для выполнения
неравенства $I(\overline{u}) \le I(u)$ в одномерном случае. Как и для монотонной перестановки, можно
считать, что $F$ зависит только от последнего аргумента, а $a$ --- только от первого.

\begin{lm}
$a$ должна удовлетворять условию $a( s ) + a( t ) \ge a( \frac{ s - t }{2} ) + a( \frac{ t - s }{2} )$ для
любых $s$ и $t$.
\end{lm}

\begin{proof}
Пусть, для определенности, $s < t$. И пусть
$a( s + z ) + a( t - z ) + 2 \delta < a( \frac{ s - t + 2 z }{2} ) + a( \frac{ t - s - 2 z }{2} )$
для $0 \le z \le \eps$, где $2 \eps < t - s$.
Построим функцию $u$: $u( x ) = 0$, если $x \le s$ или $x \ge t$; $u( x ) = 1$ при $s + \eps \le x \le t - \eps$; на
оставшихся промежутках она будет линейная с производной $\frac{1}{\eps}$ и $-\frac{1}{\eps}$ соответственно.
Тогда $\overline{u}( x ) = 0$ при $x \le \frac{ s - t }{2}$ и при $x \ge \frac{t - s}{2}$;
$\overline{u}( x ) = 1$ при $\frac{s - t}{2} + \eps \le x \le \frac{t - s}{2} - \eps$; на оставшихся двух кусках
опять линейная с теми же производными.

Тогда 
\begin{multline*}
0 \le I(u) - I(\overline{u}) =
\\ \int_s^{s + \eps} F( \frac{a( x )}{\eps} ) dx + \int_{t - \eps}^{t} F( \frac{a( x )}{\eps} ) dx -
\int_{\frac{s - t}{2}}^{\frac{s - t}{2} + \eps} F( \frac{a( x )}{\eps} ) dx -
\int_{\frac{t - s}{2} - \eps}^{\frac{t - s}{2}} F( \frac{a( x )}{\eps} ) dx =: J.
\end{multline*}
Обозначим $A = \max a$. Подберем $F( x ) = x^\alpha$ так,
чтобы $\alpha > 1$ и $x \le F( x ) \le x + \delta$ для $x \le \frac{ A }{\eps}$.
Для этого возьмем $\alpha = \log_{ \frac{A}{\eps} }(\delta + \frac{A}{\eps})$.
Тогда $$J \le \int_s^{s + \eps} \frac{a( x )}{\eps} dx + \int_{t - \eps}^{t} \frac{a( x )}{\eps} dx -
\int_{\frac{s - t}{2}}^{\frac{s - t}{2} + \eps} \frac{a( x )}{\eps} dx -
\int_{\frac{t - s}{2} - \eps}^{\frac{t - s}{2}} \frac{a( x )}{\eps} dx < 0.$$

Для $s = t$ неравенство верно по непрерывности.
\end{proof}

\begin{lm}
Если функция $a \in C^1([-1, 1])$ удовлетворяет $a( s ) + a( t ) \ge a( \frac{ s - t }{2} ) + a( \frac{ t - s }{2} )$ для
любых $s$ и $t$, то она четная.
\end{lm}

\begin{proof}
Рассмотрим функцию $b( t ) = a( s ) + a( t ) - a( \frac{ s - t }{2} ) - a( \frac{ t - s }{2} ) \ge 0$.
При $t = -s$ $b( t ) = 0$, то есть достигает минимума. Значит ее производная 
$b'( -s ) = a'( -s ) + \frac{1}{2} a'( s ) - \frac{1}{2} a'( -s ) = 0$, то есть $a'( s ) = -a'( -s )$.
Тем самым, функция четна.
\end{proof}

\begin{lm}
Если функция $a \in C([-1, 1])$ такова, что $a( s ) + a( t ) \ge a( \frac{ s - t }{2} ) + a( \frac{ t - s }{2} )$ для
любых $s$ и $t$, то она четная.
\end{lm}

\begin{proof}
Продолжим $a( x ) = a( -1 )$ при $x < -1$ и $a( x ) = a( 1 )$ при $x > 1$.
Рассмотрим усреднение функции: $a_\rho( x ) = \int_\Real w_\rho ( z ) a( x - z ) dz = \int_\Real w_\rho ( z ) a( x + z ) dz$.
Тогда 
\begin{multline*}
a_\rho( s ) + a_\rho( t ) - a_\rho( \frac{ s - t }{2} ) - a_\rho( \frac{ t - s }{2} ) =
\\ \int_\Real w_\rho ( z ) ( a( s - z ) + a( t + z ) - a( \frac{ s - t }{2} - z ) - a( \frac{ t - s }{2} + z ) ) dz \ge 0.
\end{multline*}
Значит $a_\rho$ --- четная. Кроме того, $a_\rho( x ) \to a( x )$ равномерно. Тогда
$a( x ) = \lim a_\rho( x ) = \lim a_\rho( -x ) = a( -x )$.
\end{proof}

\begin{lm}
Если четная функция $a$ удовлетворяет $a( s ) + a( t ) \ge a( \frac{ s - t }{2} ) + a( \frac{ t - s }{2} )$ для
любых $s$ и $t$, то она выпукла.
\end{lm}

\begin{proof}
Действительно, для любых $s$, $t$ $a( s ) + a( t ) = a( s ) + a( -t ) \ge 2 a( \frac{ s + t }{2} )$.
\end{proof}
